Availability
SELECT ?timevalue ?value{{
?metric rdf:type omn-monitoring-metric:Availability .
?metric omn-monitoring-data:hasMeasurementData ?data .
?data omn-monitoring-data:hasMeasurementDataValue ?value .
?data omn-monitoring-data:hasTimestamp ?timevalue
}}
limit 200
