70 START_TEST (test_filter_create)
71 {
72  /*
73  * Create an averaging filter and check that its core (generic) data
74  * structure was correctly initialized.
75  */
76  OmlFilter* f = NULL;
77  OmlFilterDef* def = NULL;
78 
79  // TBD: turn this into a loop with a check for every different type of filter.
80  f = create_filter ("avg", "avginst", OML_INT32_VALUE, NULL, 2);
81 
82  fail_if (f == NULL, "Filter creation failed for averaging filter");
83 
84  fail_unless (strcmp (f->name, "avginst") == 0, "Filter name is incorrect (%s), should be \"avginst\"", f->name);
85  fail_unless (f->output_count == 3, "Filter output width is incorrect (%d), should be 3: avg, min, max", f->output_count);
86  fail_if (f->set == NULL, "Filter set function is NULL");
87  fail_if (f->input == NULL, "Filter input function is NULL");
88  fail_if (f->output == NULL, "Filter output function is NULL");
89  fail_if (f->newwindow == NULL, "Filter newwindow function is NULL");
90  fail_if (f->meta == NULL, "Filter meta function is NULL");
91  fail_if (f->definition == NULL, "Filter definition is NULL");
92  fail_if (f->instance_data == NULL, "Filter instance data is NULL");
93 
94  def = f->definition;
95 
96  fail_unless (strcmp (def[0].name, "avg") == 0, "Filter definition error: def[0].name = %s, should be \"avg\"", def[0].name);
97  fail_unless (strcmp (def[1].name, "min") == 0, "Filter definition error: def[1].name = %s, should be \"min\"", def[1].name);
98  fail_unless (strcmp (def[2].name, "max") == 0, "Filter definition error: def[2].name = %s, should be \"max\"", def[2].name);
99 
100  fail_unless (def[0].type == OML_DOUBLE_VALUE, "Filter definition error: def[0].type = %s, should be OML_DOUBLE_VALUE", def[0].type);
101  fail_unless (def[1].type == OML_DOUBLE_VALUE, "Filter definition error: def[1].type = %s, should be OML_DOUBLE_VALUE", def[1].type);
102  fail_unless (def[2].type == OML_DOUBLE_VALUE, "Filter definition error: def[2].type = %s, should be OML_DOUBLE_VALUE", def[2].type);
103 
104  fail_unless (f->index == 2);
105  fail_unless (f->input_type == OML_INT32_VALUE);
106 
107  fail_unless (destroy_filter(f) == NULL);
108 }
109 END_TEST
