114 status_callback(SockEvtSource *source, SocketStatus status, int error, void *handle)
115 {
116  (void)error;
117  switch (status) {
118  case SOCKET_CONN_CLOSED: {
119  Client* self = (Client*)handle;
120  if (self->recv_event != NULL) {
121  socket_close (source->socket);
122  logdebug("socket '%s' closed\n", source->name);
123  eventloop_socket_remove (source); // Note: this free()'s source!
124  }
125 
126  /* Signal the sender thread for this client that the client disconnected */
127  pthread_mutex_lock (&self->mutex);
128  self->state = C_DISCONNECTED;
129  pthread_cond_signal (&self->condvar);
130  pthread_mutex_unlock (&self->mutex);

132  default:

134  }

166 /*
167  * A hack to work around Mac OSX's broken implementation of poll(2),
168  * which does not allow polling stdin. This hack dup()'s stdin to
169  * another file descriptor and creates a pipe(2) which it then
170  * dup2()'s into the normal stdin file descriptor. If this succeeds,
171  * it then spawns a thread whose only function is to read(2) from the
172  * original stdin and write into the pipe. The read end of the pipe,
173  * which is now file descriptor 0, is then useable in a call to
174  * poll(2).
175  */
176 void*
177 prepare_stdin (void *handle)
178 {
179  (void)handle;
180 #ifdef __APPLE__
181  static int thread_started = 0;
182  static int stdin_dup;
183  static int stdin_pipe[2];
184  static pthread_t self;
185  if (!thread_started) {
186  int result;
187  result = pipe (stdin_pipe);
188  if (result) {
189  logerror ("Could not create pipe for stdin duplication: %s\n", strerror(errno));
190  exit(EXIT_FAILURE);
191  }
192 
193  stdin_dup = dup(fileno(stdin));
194  dup2(stdin_pipe[0], fileno(stdin));
195 
196  thread_started = 1;
197  result = pthread_create (&self, NULL, prepare_stdin, NULL);
198  if (result == -1) {
199  logerror ("Error creating thread for reading stdin: %s\n", strerror (errno));
200  exit (EXIT_FAILURE);
201  }
202  } else {
203  char buf [512];
204  while (1) {
205  ssize_t count = read (stdin_dup, buf, 512);
206  if (count < 0)
207  logerror ("Error reading from stdin: %s\n", strerror(errno));
208  if (count <= 0)
209  return NULL;
210  write (stdin_pipe[1], buf, count);
211  }
212  }
213 #endif
214  return NULL;
215 }

217 void
218 stdin_handler(SockEvtSource* source, void* handle, void* buf, int buf_size)
219 {
220  Session *proxy = (Session*)handle;
221  char command[80];
222  (void)source;
223  strncpy (command, buf, 80);
224 
225  if (buf_size < 80 && command[buf_size-1] == '\n')
226  command[buf_size-1] = '\0';
227 
228  printf ("Received command: %s\n", command);
229  logdebug ("Received command: %s\n", command);
230 
231  if ((strcmp (command, "OMLPROXY-RESUME") == 0) ||
232  (strcmp (command, "RESUME") == 0)) {
233  proxy->state = ProxyState_SENDING;
234  } else if (strcmp (command, "OMLPROXY-STOP") == 0 ||
235  strcmp (command, "STOP") == 0) {
236  proxy->state = ProxyState_STOPPED;
237  } else if (strcmp (command, "OMLPROXY-PAUSE") == 0 ||
238  strcmp (command, "PAUSE") == 0) {
239  proxy->state = ProxyState_PAUSED;
240  }
241 
242  /*
243  * If we're in sending state, wake up the client sender threads
244  * so that they will start sending to the downstream server. We do
245  * this even if the state was already ProxyState_SENDING because
246  * some of the clients might have dropped back to idle due to
247  * disconnection from the upstream server.
248  */
249  if (session->state == ProxyState_SENDING) {
250  Client *current = session->clients;
251  while (current) {
252  pthread_mutex_lock (&current->mutex);
253  pthread_cond_signal (&current->condvar);
254  pthread_mutex_unlock (&current->mutex);
255  current = current->next;
256  }
257  }
258 }
