570 marshal_value(MBuffer* mbuf, OmlValueT val_type, OmlValueU* val)
571 {
572  switch (val_type) {
573  case OML_LONG_VALUE: {
574  long v = oml_value_clamp_long (omlc_get_long(*val));
575  uint32_t uv = (uint32_t)v;
576  uint32_t nv = htonl(uv);
577  uint8_t buf[LONG_T_SIZE+1];
578 
579  buf[0] = LONG_T;
580  memcpy(&buf[1], &nv, sizeof (nv));
581 
582  logdebug3("Marshalling long %ld\n", nv);
583  int result = mbuf_write (mbuf, buf, LENGTH (buf));
584  if (result == -1) {
585  logerror("Failed to marshal OML_LONG_VALUE (mbuf_write())\n");
586  mbuf_reset_write (mbuf);
587  return 0;
588  }
589  break;
590  }
591  case OML_INT32_VALUE:
592  case OML_UINT32_VALUE:
593  case OML_INT64_VALUE:
594  case OML_UINT64_VALUE: {
595  uint8_t buf[UINT64_T_SIZE+1]; // Max integer size
596  uint32_t uv32;
597  uint32_t nv32;
598  uint64_t uv64;
599  uint64_t nv64;
600  uint8_t *p_nv;
601 
602  if (oml_size_map[val_type] == 4)
603  {
604  uv32 = omlc_get_uint32(*val);
605  nv32 = htonl(uv32);
606  p_nv = (uint8_t*)&nv32;
607  logdebug3("Marshalling %s %" PRIu32 "\n", oml_type_to_s(val_type), uv32);
608  } else {
609  uv64 = omlc_get_uint64(*val);
610  nv64 = htonll(uv64);
611  p_nv = (uint8_t*)&nv64;
612  logdebug3("Marshalling %s %" PRIu64 "\n", oml_type_to_s(val_type), uv64);
613  }
614 
615  buf[0] = oml_type_map[val_type];
616  memcpy(&buf[1], p_nv, oml_size_map[val_type]);
617 
618  int result = mbuf_write (mbuf, buf, oml_size_map[val_type] + 1);
619  if (result == -1)
620  {
621  logerror("Failed to marshal %s value (mbuf_write())\n",
622  oml_type_to_s (val_type));
623  mbuf_reset_write (mbuf);
624  return 0;
625  }
626  break;
627  }
628  case OML_DOUBLE_VALUE: {
629  uint8_t type = DOUBLE_T;
630  double v = omlc_get_double(*val);
631  int exp;
632  double mant = frexp(v, &exp);
633  int8_t nexp = (int8_t)exp;
634  logdebug3("Marshalling double %f\n", v);
635  if (isnan(v)) {
636  type = DOUBLE_NAN;
637  nexp = 0;
638  mant = 0;
639  } else if (nexp != exp) {
640  logerror("Double number '%lf' is out of bounds, sending NaN\n", v);
641  type = DOUBLE_NAN;
642  nexp = 0;
643  mant = 0;
644  }
645  int32_t imant = (int32_t)(mant * (1 << BIG_L));
646  uint32_t nmant = htonl(imant);
647 
648  uint8_t buf[6] = { type, 0, 0, 0, 0, nexp };
649 
650  memcpy(&buf[1], &nmant, sizeof (nmant));
651 
652  int result = mbuf_write (mbuf, buf, LENGTH (buf));
653 
654  if (result == -1)
655  {
656  logerror("Failed to marshal OML_DOUBLE_VALUE (mbuf_write())\n");
657  mbuf_reset_write (mbuf);
658  return 0;
659  }
660  break;
661  }
662  case OML_STRING_VALUE: {
663  char* str = omlc_get_string_ptr(*val);
664 
665  if (str == NULL) {
666  str = "";
667  logdebug("Attempting to send a NULL string; sending empty string instead\n");
668  }
669 
670  size_t len = strlen(str);
671  if (len > STRING_T_MAX_SIZE) {
672  logerror("Truncated string '%s'\n", str);
673  len = STRING_T_MAX_SIZE;
674  }
675 
676  logdebug3("Marshalling string '%s' of length %d\n", str, len);
677  uint8_t buf[2] = { STRING_T, (uint8_t)(len & 0xff) };
678  int result = mbuf_write (mbuf, buf, LENGTH (buf));
679  if (result == -1) {
680  logerror("Failed to marshal OML_STRING_VALUE type and length (mbuf_write())\n");
681  mbuf_reset_write (mbuf);
682  return 0;
683  }
684 
685  result = mbuf_write (mbuf, (uint8_t*)str, len);
686 
687  if (result == -1)
688  {
689  logerror("Failed to marshal OML_STRING_VALUE (mbuf_write())\n");
690  mbuf_reset_write (mbuf);
691  return 0;
692  }
693  break;
694  }
695  case OML_BLOB_VALUE: {
696  int result = 0;
697  void *blob = omlc_get_blob_ptr(*val);
698  size_t length = omlc_get_blob_length(*val);
699  if (blob == NULL || length == 0) {
700  logdebug ("Attempting to send NULL or empty blob; blob of length 0 will be sent\n");
701  length = 0;
702  }
703 
704  uint8_t buf[5] = { BLOB_T, 0, 0, 0, 0 };
705  size_t n_length = htonl (length);
706  memcpy (&buf[1], &n_length, 4);
707 
708  logdebug3("Marshalling blob of size %d\n", length);
709  result = mbuf_write (mbuf, buf, sizeof (buf));
710 
711  if (result == -1) {
712  logerror ("Failed to marshall OML_BLOB_VALUE type and length (mbuf_write())\n");
713  mbuf_reset_write (mbuf);
714  return 0;
715  }
716 
717  result = mbuf_write (mbuf, blob, length);
718 
719  if (result == -1) {
720  logerror ("Failed to marshall %d bytes of OML_BLOB_VALUE data\n", length);
721  mbuf_reset_write (mbuf);
722  return 0;
723  }
724  break;
725  }
726 
727  case OML_GUID_VALUE: {
728  /* FIXME: Wrap with UINT64 marshalling, just change the type */
729  uint64_t nv64;
730  uint8_t buf[GUID_T_SIZE+1];
731  buf[0] = GUID_T;
732  nv64 = htonll(omlc_get_guid(*val));
733  memcpy(&buf[1], &nv64, sizeof(nv64));
734  logdebug3("Marshalling GUID %" PRIu64 "\n", nv64);
735  if (-1 == mbuf_write(mbuf, buf, LENGTH(buf))) {
736  logerror("Failed to marshal OML_GUID_VALUE (mbuf_write())\n");
737  mbuf_reset_write(mbuf);
738  return 0;
739  }
740  break;
741  }
742 
743  case OML_BOOL_VALUE: {
744  uint8_t buf;
745  if (!omlc_get_bool(*val)) {
746  buf = BOOL_FALSE_T;
747  } else {
748  buf = BOOL_TRUE_T;
749  }
750  logdebug3("Marshalling boolean %d\n", BOOL_TRUE_T == buf);
751  if (-1 == mbuf_write(mbuf, &buf, 1)) {
752  logerror("Failed to marshal OML_BOOL_VALUE (mbuf_write())\n");

755  }

757  }

759  case OML_VECTOR_INT32_VALUE:
760  case OML_VECTOR_UINT32_VALUE: {

855 int
856 marshal_finalize(MBuffer* mbuf)
857 {
858  uint8_t* buf = mbuf_message (mbuf);
859  OmlBinMsgType type = marshal_get_msgtype (mbuf);
860  size_t len = mbuf_message_length (mbuf);
861 
862  if (len > UINT32_MAX) {
863  logwarn("Message length %d longer than maximum packet length (%d); "
864  "packet will be truncated\n",
865  len, UINT32_MAX);
866  len = UINT32_MAX;
867  }
868 
869  if (type == OMB_DATA_P && len > UINT16_MAX) {
870  /*
871  * We assumed a short packet, but there is too much data, so we
872  * have to shift the whole buffer down by 2 bytes and convert to a
873  * long packet.
874  */
875  uint8_t s[2] = {0};
876  /* Put some padding in the buffer to make sure it has room, and maintains its invariants */
877  mbuf_write (mbuf, s, sizeof (s));
878  memmove (&buf[PACKET_HEADER_SIZE+2], &buf[PACKET_HEADER_SIZE],
879  len - PACKET_HEADER_SIZE);
880  len += 2;
881  buf[2] = type = OMB_LDATA_P;
882  }
883 
884 
885  switch (type) {
886  case OMB_DATA_P:
887  len -= PACKET_HEADER_SIZE; // Data length minus header
888  uint16_t nlen16 = htons (len);
889  memcpy (&buf[3], &nlen16, sizeof (nlen16));
890  break;
891  case OMB_LDATA_P:
892  len -= PACKET_HEADER_SIZE + 2; // Data length minus header
893  uint32_t nlen32 = htonl (len); // pure data length
894  memcpy (&buf[3], &nlen32, sizeof (nlen32));
895  break;
896  }
897 
898  return 1;
899 }

936  }

938  header->type = (OmlBinMsgType)header_str[2];
939 
940  if (header->type == OMB_DATA_P) {
941  // Read 2 more bytes of the length field
942  uint16_t nv16 = 0;
943  result = mbuf_read (mbuf, (uint8_t*)&nv16, sizeof (uint16_t));
944  if (result == -1) {
945  n = mbuf_rd_remaining (mbuf) - 2;
946  mbuf_reset_read (mbuf);
947  return n;
948  }
949  header->length = (int)ntohs (nv16);
950  } else if (header->type == OMB_LDATA_P) {
951  // Read 4 more bytes of the length field
952  uint32_t nv32 = 0;
953  result = mbuf_read (mbuf, (uint8_t*)&nv32, sizeof (uint32_t));
954  if (result == -1) {
955  n = mbuf_rd_remaining (mbuf) - 4;
956  mbuf_reset_read (mbuf);
957  return n;
958  }
959  header->length = (int)ntohl (nv32);
960  } else {
961  logwarn ("Unknown packet type %d\n", (int)header->type);
962  return 0;
963  }
964 
965  int extra = mbuf_rd_remaining (mbuf) - header->length;
966  if (extra < 0) {
967  mbuf_reset_read (mbuf);
968  return extra;
969  }
970 
971  result = mbuf_read (mbuf, stream_header_str, LENGTH (stream_header_str));
972  if (result == -1) {
973  n = mbuf_rd_remaining (mbuf) - LENGTH (stream_header_str);
974  mbuf_reset_read (mbuf);
975  return n;
976  }
977 
978  header->values = (int)stream_header_str[0];
979  header->stream = (int)stream_header_str[1];
980 
981  if (unmarshal_typed_value (mbuf, "seq-no", OML_INT32_VALUE, &seqno) == -1)
982  return 0;
983 
984  if (unmarshal_typed_value (mbuf, "timestamp", OML_DOUBLE_VALUE, &timestamp) == -1)
985  return 0;
986 
987  header->seqno = omlc_get_int32(*oml_value_get_value(&seqno));
988  header->timestamp = omlc_get_double(*oml_value_get_value(&timestamp));
989 
990  oml_value_reset(&seqno);
991  oml_value_reset(&timestamp);
992 
993  return 1;
994 }

1059 int
1060 unmarshal_value(MBuffer *mbuf, OmlValue *value)
1061 {
1062  if (mbuf_rd_remaining(mbuf) == 0) {
1063  logerror("Tried to unmarshal a value from the buffer, but didn't receive enough data to do that\n");
1064  return 0;
1065  }
1066 
1067  int type = mbuf_read_byte (mbuf);
1068  if (type == -1) return 0;
1069 
1070  switch (type) {
1071  case LONG_T: {
1072  uint8_t buf [LONG_T_SIZE];
1073 
1074  if (mbuf_read (mbuf, buf, LENGTH (buf)) == -1)
1075  {
1076  logerror("Failed to unmarshal OML_LONG_VALUE; not enough data?\n");
1077  return 0;
1078  }
1079 
1080  uint32_t hv = ntohl(*((uint32_t*)buf));
1081  int32_t v = (int32_t)(hv);
1082 
1083  /*
1084  * The server no longer needs to know about OML_LONG_VALUE, as the
1085  * marshalling process now maps OML_LONG_VALUE into OML_INT32_VALUE
1086  * (by truncating to [INT_MIN, INT_MAX]. Therefore, unmarshall a
1087  * LONG_T value into an OML_INT32_VALUE object.
1088  */
1089  oml_value_set_type(value, OML_INT32_VALUE);
1090  omlc_set_int32(*oml_value_get_value(value), v);
1091  break;
1092  }
1093  case INT32_T:
1094  case UINT32_T:
1095  case INT64_T:
1096  case UINT64_T: {
1097  uint8_t buf [UINT64_T_SIZE]; // Maximum integer size
1098  OmlValueT oml_type = protocol_type_map[type];
1099  if (mbuf_read (mbuf, buf, protocol_size_map[type]) == -1) {
1100  logerror("Failed to unmarshall %d value; not enough data?\n", type);
1101  return 0;
1102  }
1103  oml_value_set_type(value, oml_type);
1104  switch (type) {
1105  case INT32_T:
1106  omlc_set_int32(*oml_value_get_value(value), ntohl(*((int32_t*)buf)));
1107  logdebug3("Unmarshalled %s %" PRId32 "\n", oml_type_to_s(oml_type), omlc_get_int32(*oml_value_get_value(value)));
1108  break;
1109 
1110  case UINT32_T:
1111  omlc_set_uint32(*oml_value_get_value(value), ntohl(*((uint32_t*)buf)));
1112  logdebug3("Unmarshalled %s %" PRIu32 "\n", oml_type_to_s(oml_type), omlc_get_uint32(*oml_value_get_value(value)));
1113  break;
1114 
1115  case INT64_T:
1116  omlc_set_int64(*oml_value_get_value(value), ntohll(*((int64_t*)buf)));
1117  logdebug3("Unmarshalled %s %" PRId64 "\n", oml_type_to_s(oml_type), omlc_get_int64(*oml_value_get_value(value)));
1118  break;
1119 
1120  case UINT64_T:
1121  omlc_set_uint64(*oml_value_get_value(value), ntohll(*((uint64_t*)buf)));
1122  logdebug3("Unmarshalled %s %" PRIu64 "\n", oml_type_to_s(oml_type), omlc_get_uint64(*oml_value_get_value(value)));
1123  break;
1124 
1125  default:
1126  logerror("Integer morphed, something magic has just happened\n");
1127  return 0;
1128  }
1129  break;
1130  }
1131  case DOUBLE_T: {
1132  uint8_t buf [DOUBLE_T_SIZE];
1133  OmlValueT oml_type = protocol_type_map[type];
1134  if (mbuf_read (mbuf, buf, LENGTH (buf)) == -1)
1135  {
1136  logerror("Failed to unmarshal OML_DOUBLE_VALUE; not enough data?\n");
1137  return 0;
1138  }
1139 
1140  int hmant = (int)ntohl(*((uint32_t*)buf));
1141  double mant = hmant * 1.0 / (1 << BIG_L);
1142  int exp = (int8_t) buf[4];
1143  double v = ldexp(mant, exp);
1144  oml_value_set_type(value, oml_type);
1145  omlc_set_double(*oml_value_get_value(value), v);
1146  logdebug3("Unmarshalled double %f\n", omlc_get_double(*oml_value_get_value(value)));
1147  break;
1148  }
1149  case DOUBLE_NAN: {
1150  OmlValueT oml_type = protocol_type_map[type];
1151  mbuf_read_skip(mbuf, DOUBLE_T_SIZE); /* The data is irrelevant */
1152  oml_value_set_type(value, oml_type);
1153  omlc_set_double(*oml_value_get_value(value), NAN);
1154  logdebug("Received NaN\n");
1155  break;
1156  }
1157  case STRING_T: {
1158  int len = 0;
1159  uint8_t buf [STRING_T_MAX_SIZE];
1160 
1161  len = mbuf_read_byte (mbuf);
1162 
1163  if (len == -1 || mbuf_read (mbuf, buf, len) == -1)
1164  {
1165  logerror("Failed to unmarshal OML_STRING_VALUE; not enough data?\n");
1166  return 0;
1167  }
1168 
1169  oml_value_set_type(value, OML_STRING_VALUE);
1170  omlc_set_string_copy(*oml_value_get_value(value), buf, len);
1171  logdebug3("Unmarshalled string '%s' of length %d\n", omlc_get_string_ptr(*oml_value_get_value(value)), len);
1172  break;
1173  }
1174  case BLOB_T: {
1175  uint32_t n_len;
1176 
1177  if (mbuf_read (mbuf, (uint8_t*)&n_len, 4) == -1) {
1178  logerror ("Failed to unmarshal OML_BLOB_VALUE length field; not enough data?\n");
1179  return 0;
1180  }
1181 
1182  size_t len = ntohl (n_len);
1183  size_t remaining = mbuf_rd_remaining (mbuf);
1184 
1185  if (len > remaining) {
1186  logerror ("Failed to unmarshal OML_BLOB_VALUE data: not enough data available "
1187  "(wanted %d, but only have %d bytes\n",
1188  len, remaining);
1189  return 0;
1190  }
1191 
1192  void *ptr = mbuf_rdptr (mbuf);
1193  oml_value_set_type(value, OML_BLOB_VALUE);
1194  omlc_set_blob (*oml_value_get_value(value), ptr, len); /*XXX*/
1195  logdebug3("Unmarshalled blob of size %d\n", len);
1196  mbuf_read_skip (mbuf, len);
1197  break;
1198  }
1199 
1200  case GUID_T: {
1201  uint64_t nv64;
1202  uint8_t buf[GUID_T_SIZE];
1203  if(mbuf_read(mbuf, buf, GUID_T_SIZE) == -1) {
1204  logerror("Failed to unmarshall OML_GUID_VALUE data; not enough data?\n");
1205  return 0;
1206  }
1207  memcpy(&nv64, buf, sizeof(nv64));
1208  oml_value_set_type(value, OML_GUID_VALUE);
1209  omlc_set_guid(*oml_value_get_value(value), ntohll(nv64));
1210  logdebug3("Unmarshalled GUID %" PRIu64 "\n", omlc_get_guid(*oml_value_get_value(value)));
1211  break;
1212  }
1213 
1214  case BOOL_FALSE_T:
1215  case BOOL_TRUE_T:
1216  oml_value_set_type(value, OML_BOOL_VALUE);
1217  omlc_set_bool(*oml_value_get_value(value),
1218  (type == BOOL_TRUE_T)?OMLC_BOOL_TRUE:OMLC_BOOL_FALSE);
1219  logdebug3("Unmarshalled boolean %d\n", OMLC_BOOL_TRUE == omlc_get_bool(*oml_value_get_value(value)));
1220  break;
1221 
1222  case VECTOR_T: {
1223  uint16_t i, nof_elts;
1224  int type = mbuf_read_byte(mbuf);
1225  if(-1 == type) {
1226  logerror("%s(): failed to unmarshall VECTOR_T length\n", __func__);
1227  return 0;
1228  }
1229  if(mbuf_read(mbuf,(uint8_t*)(&nof_elts), sizeof(nof_elts)) == -1) {
1230  logerror("%s(): failed to unmarshall VECTOR_T length\n", __func__);
1231  return 0;
1232  }
1233  nof_elts = ntohs(nof_elts);
1234 
1235  OmlValueT oml_type = vector_type_map[type];
1236  OmlValueU *v = oml_value_get_value(value);
1237  switch(type) {
1238  case INT32_T:
1239  case UINT32_T: {
1240  size_t bytes = nof_elts * sizeof(uint32_t);
1241  uint32_t *elts = oml_calloc(nof_elts, sizeof(uint32_t));
1242  if(mbuf_read(mbuf, (uint8_t*)(elts), nof_elts * sizeof(uint32_t)) == -1) {
1243  logerror("%s(): failed to unmarshall OML_VECTOR_(U)INT32_VALUE\n", __func__);
1244  return 0;
1245  }
1246  for(i = 0; i < nof_elts; i++)
1247  elts[i] = ntohl(elts[i]);
1248  oml_value_set_type(value, oml_type);
1249  omlc_set_vector_ptr(*v, elts);
1250  omlc_set_vector_length(*v, bytes);
1251  omlc_set_vector_size(*v, bytes);
1252  omlc_set_vector_nof_elts(*v, nof_elts);
1253  omlc_set_vector_elt_size(*v, sizeof(uint32_t));
1254  break;
1255  }
1256  case INT64_T:
1257  case UINT64_T:
1258  case DOUBLE64_T: {
1259  size_t bytes = nof_elts * sizeof(uint64_t);
1260  uint64_t *elts = oml_calloc(nof_elts, sizeof(uint64_t));
1261  if(mbuf_read(mbuf, (uint8_t*)(elts), nof_elts * sizeof(uint64_t)) == -1) {
1262  logerror("%s(): failed to unmarshall OML_VECTOR_(U)INT64_VALUE\n", __func__);
1263  return 0;
1264  }
1265  for(i = 0; i < nof_elts; i++)
1266  elts[i] = ntohll(elts[i]);
1267  oml_value_set_type(value, oml_type);
1268  omlc_set_vector_ptr(*v, elts);
1269  omlc_set_vector_length(*v, bytes);
1270  omlc_set_vector_size(*v, bytes);
1271  omlc_set_vector_nof_elts(*v, nof_elts);
1272  omlc_set_vector_elt_size(*v, sizeof(uint64_t));
1273  break;
1274  }
1275  case BOOL_T: {
1276  uint8_t y[nof_elts];
1277  size_t bytes = nof_elts * sizeof(bool);
1278  bool *elts = oml_calloc(nof_elts, sizeof(bool));
1279  if(mbuf_read(mbuf, y, nof_elts) == -1) {
1280  logerror("%s(): failed to unmarshall OML_VECTOR_BOOL_VALUE\n", __func__);
1281  return 0;
1282  }
1283  for(i = 0; i < nof_elts; i++)
1284  elts[i] = ((BOOL_TRUE_T == y[i]) ? true : false);
1285  oml_value_set_type(value, oml_type);
1286  omlc_set_vector_ptr(*v, elts);
1287  omlc_set_vector_length(*v, bytes);
1288  omlc_set_vector_size(*v, bytes);
1289  omlc_set_vector_nof_elts(*v, nof_elts);
1290  omlc_set_vector_elt_size(*v, sizeof(bool));
1291  break;
1292  }
1293  default:
1294  logerror("%s(): bad type for array (t=%d)\n", __func__, type);
1295  break;
1296  }
1297  break;
1298  }
1299 
1300  default:
1301  logerror("%s: Unsupported value type '%d'\n", __FUNCTION__, type);
1302  return 0;
1303  }
1304 
1305  return 1;
1306 }
