132  /*
133  * The man page says access(2) should be avoided because it creates
134  * a race condition between calls to access(2) and open(2) where an
135  * attacker could swap the underlying file for a link to a file that
136  * the unprivileged user does not have permissions to access, which
137  * the effective user does. We don't use the access/open sequence
138  * here. We check that we can create files in the
139  * sqlite_database_dir directory, and fail if not (as the server
140  * won't be able to do useful work otherwise).
141  *
142  * An attacker could potentially change the underlying file to a
143  * link and still cause problems if oml2-server is run as root or
144  * setuid root, but this check must happen after drop_privileges()
145  * is called, so if oml2-server is not run with superuser privileges
146  * such an attack would still fail.
147  *
148  * oml2-server should not be run as root or setuid root in any case.
149  */
150  if (access (sqlite_database_dir, R_OK | W_OK | X_OK) == -1) {
151  logerror ("sqlite: Can't access SQLite database directory %s: %s\n",
152  sqlite_database_dir, strerror (errno));
153  return -1;
154  }
155 
156  loginfo ("sqlite: Creating SQLite3 databases in %s\n", sqlite_database_dir);
157 
158  return 0;
159 }

324  table->schema->name);

333  db->name, table->schema->name);

335  }
336

339  logerror("sqlite:%s: Could not prepare statement '%s': %s\n",
340  db->name, mstring_buf(insert), sqlite3_errmsg(sq3db->conn));

342  }

407 MString*
408 sq3_prepare (Database *db, DbTable* table)
409 {
410  MString* mstr = mstring_create ();
411  int n = 0, i = 0;
412  int max = table->schema->nfields;
413  char *pvar;
414 
415  if (max <= 0) {
416  logerror ("%d: Trying to insert 0 values into table %s\n",
417  db->backend_name, table->schema->name);
418  goto fail_exit;
419  }
420 
421  if (mstr == NULL) {
422  logerror("%d: Failed to create managed string for preparing SQL INSERT statement\n",
423  db->backend_name);
424  goto fail_exit;
425  }
426 
427  /* Build SQL "INSERT INTO" statement */
428  n += mstring_sprintf (mstr,
429  "INSERT INTO \"%s\" (\"oml_sender_id\", \"oml_seq\", \"oml_ts_client\", \"oml_ts_server\"",
430  table->schema->name);
431 
432  /* Add specific column names */
433  for (i=0; i<max; i++) {
434  n += mstring_sprintf (mstr, ", \"%s\"", table->schema->fields[i].name);
435  }
436 
437  /* Close column names, and add variables for the prepared statement */
438  pvar = db->prepared_var(db, 1);
439  n += mstring_sprintf (mstr, ") VALUES (%s", pvar);
440  oml_free(pvar); /* XXX: Not really efficient, but we only do this rarely */
441 
442  max += 4; /* Number of metadata columns we want to be able to insert */
443  for (i=2; i<=max; i++) {
444  pvar = db->prepared_var(db, i);
445  n += mstring_sprintf (mstr, ", %s", pvar);
446  oml_free(pvar);
447  }
448 
449  /* We're done */
450  n += mstring_cat (mstr, ");");
451 
452  logdebug2("%s:%s: Prepared insert statement for table %s: %s\n",
453  db->backend_name, db->name, table->schema->name, mstring_buf(mstr));
454 
455  if (n != 0) {
456  /* mstring_* return -1 on error, with no error, the sum in n should be 0 */
457  goto fail_exit;
458  }
459  return mstr;
460 
461  fail_exit:
462  if (mstr) mstring_delete (mstr);
463  return NULL;
464 }

470 static int
471 sq3_insert(Database *db, DbTable *table, int sender_id, int seq_no, double time_stamp, OmlValue *values, int value_count)
472 {
473  Sq3DB* sq3db = (Sq3DB*)db->handle;
474  Sq3Table* sq3table = (Sq3Table*)table->handle;
475  int i;
476  double time_stamp_server;
477  sqlite3_stmt* stmt = sq3table->insert_stmt;
478  char *json = NULL;
479  ssize_t json_sz;
480  struct timeval tv;
481  gettimeofday(&tv, NULL);
482  time_stamp_server = tv.tv_sec - db->start_time + 0.000001 * tv.tv_usec;
483 
484  if (tv.tv_sec > sq3db->last_commit) {
485  if (dba_reopen_transaction (db) == -1) {
486  return -1;
487  }
488  sq3db->last_commit = tv.tv_sec;
489  }
490 
491  // o_log(O_LOG_DEBUG2, "sq3_insert(%s): insert row %d \n",
492  // table->schema->name, seq_no);
493 
494  if (sqlite3_bind_int(stmt, 1, sender_id) != SQLITE_OK) {
495  logerror("sqlite:%s: Could not bind 'oml_sender_id' in table '%s': %s\n",
496  db->name, table->schema->name,
497  sqlite3_errmsg(sq3db->conn));
498  }
499  if (sqlite3_bind_int(stmt, 2, seq_no) != SQLITE_OK) {
500  logerror("sqlite:%s: Could not bind 'oml_seq' in table '%s': %s\n",
501  db->name, table->schema->name,
502  sqlite3_errmsg(sq3db->conn));
503  }
504  if (sqlite3_bind_double(stmt, 3, time_stamp) != SQLITE_OK) {
505  logerror("sqlite:%s: Could not bind 'oml_ts_client' in table '%s': %s\n",
506  db->name, table->schema->name,
507  sqlite3_errmsg(sq3db->conn));
508  }
509  if (sqlite3_bind_double(stmt, 4, time_stamp_server) != SQLITE_OK) {
510  logerror("sqlite:%s: Could not bind 'oml_ts_server' in table '%s': %s\n",
511  db->name, table->schema->name,
512  sqlite3_errmsg(sq3db->conn));
513  }
514 
515  OmlValue* v = values;
516  struct schema *schema = table->schema;
517  if (schema->nfields != value_count) {
518  logerror ("sqlite:%s: Failed to insert %d values into table '%s' with %d columns\n",
519  db->name, value_count, table->schema->name, schema->nfields);
520  sqlite3_reset (stmt);
521  return -1;
522  }
523  for (i = 0; i < schema->nfields; i++, v++) {
524  if (oml_value_get_type(v) != schema->fields[i].type) {
525  const char *expected = oml_type_to_s (schema->fields[i].type);
526  const char *received = oml_type_to_s (oml_value_get_type(v));
527  logerror("sqlite:%s: Value %d type mismatch for table '%s'\n", db->name, i, table->schema->name);
528  logdebug("sqlite:%s: -> Column name='%s', type=%s, but trying to insert a %s\n",
529  db->name, schema->fields[i].name, expected, received);
530  sqlite3_reset (stmt);
531  return -1;
532  }
533  int res;
534  int idx = i + 5;
535  switch (schema->fields[i].type) {
536  case OML_DOUBLE_VALUE:
537  res = sqlite3_bind_double(stmt, idx, omlc_get_double(*oml_value_get_value(v)));
538  break;
539  case OML_LONG_VALUE:
540  res = sqlite3_bind_int(stmt, idx, (int)omlc_get_long(*oml_value_get_value(v)));
541  break;
542 
543  case OML_INT32_VALUE:
544  res = sqlite3_bind_int(stmt, idx, (int32_t)omlc_get_int32(*oml_value_get_value(v)));
545  break;
546  case OML_UINT32_VALUE:
547  res = sqlite3_bind_int(stmt, idx, (uint32_t)omlc_get_uint32(*oml_value_get_value(v)));
548  break;
549  case OML_INT64_VALUE:
550  res = sqlite3_bind_int64(stmt, idx, (int64_t)omlc_get_int64(*oml_value_get_value(v)));
551  break;
552  case OML_UINT64_VALUE:
553  if (omlc_get_uint64(*oml_value_get_value(v)) > (uint64_t)9223372036854775808ull) {
554  logwarn("sqlite:%s: Trying to store value %" PRIu64 " (>2^63) in column '%s' of table '%s', this might lead to a loss of resolution\n",
555  db->name, (uint64_t)omlc_get_uint64(*oml_value_get_value(v)), schema->fields[i].name, table->schema->name);
556  }
557 
558  res = sqlite3_bind_int64(stmt, idx, (uint64_t)omlc_get_uint64(*oml_value_get_value(v)));
559  break;
560 
561  case OML_STRING_VALUE:
562  res = sqlite3_bind_text (stmt, idx, omlc_get_string_ptr(*oml_value_get_value(v)),
563  -1, SQLITE_TRANSIENT);
564  break;
565 
566  case OML_BLOB_VALUE:
567  res = sqlite3_bind_blob (stmt, idx,
568  omlc_get_blob_ptr(*oml_value_get_value(v)),
569  omlc_get_blob_length(*oml_value_get_value(v)),
570  SQLITE_TRANSIENT);
571  break;
572 
573  case OML_GUID_VALUE:
574  if(omlc_get_guid(*oml_value_get_value(v)) != UINT64_C(0)) {
575  res = sqlite3_bind_int64(stmt, idx, (int64_t)(omlc_get_guid(*oml_value_get_value(v))));
576  } else {
577  res = sqlite3_bind_null(stmt, idx);
578  }
579  break;
580 
581  case OML_BOOL_VALUE:
582  res = sqlite3_bind_int(stmt, idx, (int)omlc_get_bool(*oml_value_get_value(v)));
583  break;
584 
585  case OML_VECTOR_DOUBLE_VALUE:
586  json = NULL;
587  json_sz = vector_double_to_json(v->value.vectorValue.ptr, v->value.vectorValue.nof_elts, &json);
588  if(-1 != json_sz)
589  res = sqlite3_bind_text(stmt, idx, json, json_sz, oml_free);
590  else
591  res = sqlite3_bind_null(stmt, idx);
592  break;
593  case OML_VECTOR_INT32_VALUE:
594  json = NULL;
595  json_sz = vector_int32_to_json(v->value.vectorValue.ptr, v->value.vectorValue.nof_elts, &json);
596  if(-1 != json_sz)
597  res = sqlite3_bind_text(stmt, idx, json, json_sz, oml_free);
598  else
599  res = sqlite3_bind_null(stmt, idx);
600  break;
601  case OML_VECTOR_UINT32_VALUE:
602  json = NULL;
603  json_sz = vector_uint32_to_json(v->value.vectorValue.ptr, v->value.vectorValue.nof_elts, &json);
604  if(-1 != json_sz)
605  res = sqlite3_bind_text(stmt, idx, json, json_sz, oml_free);
606  else
607  res = sqlite3_bind_null(stmt, idx);
608  break;
609  case OML_VECTOR_INT64_VALUE:
610  json = NULL;
611  json_sz = vector_int64_to_json(v->value.vectorValue.ptr, v->value.vectorValue.nof_elts, &json);
612  if(-1 != json_sz)
613  res = sqlite3_bind_text(stmt, idx, json, json_sz, oml_free);
614  else
615  res = sqlite3_bind_null(stmt, idx);
616  break;

626  json = NULL;
627  json_sz = vector_bool_to_json(v->value.vectorValue.ptr, v->value.vectorValue.nof_elts, &json);
628  if(-1 != json_sz)
629  res = sqlite3_bind_text(stmt, idx, json, json_sz, oml_free);
630  else
631  res = sqlite3_bind_null(stmt, idx);
632  break;
633  default:
634  logerror("sqlite:%s: Unknown type %d in col '%s' of table '%s; this is probably a bug'\n",
635  db->name, schema->fields[i].type, schema->fields[i].name, table->schema->name);
636  sqlite3_reset (stmt);
637  return -1;
638  }
639  if (res != SQLITE_OK) {
640  logerror("sqlite:%s: Could not bind column '%s': %s\n",
641  db->name, schema->fields[i].name, sqlite3_errmsg(sq3db->conn));
642  sqlite3_reset (stmt);
643  return -1;
644  }
645  }
646 
647  if (sqlite3_step(stmt) != SQLITE_DONE) {
648  logerror("sqlite:%s: Could not step SQL statement: %s\n",
649  db->name, sqlite3_errmsg(sq3db->conn));
650  sqlite3_reset(stmt);
651  return -1;
652  }
653  return sqlite3_reset(stmt);
654 }

860 static TableDescr*
861 sq3_get_table_list (Database *database, int *num_tables)
862 {
863  Sq3DB *self = database->handle;
864  TableDescr *tables = NULL, *t = NULL;
865  const char *table_stmt = "SELECT name FROM sqlite_master WHERE type='table' ORDER BY name;";
866  const char *tablename, *meta;
867  const char *schema_stmt = "SELECT value FROM _experiment_metadata WHERE key='table_%s';";
868  MString *schema_stmt_ms = NULL;
869  sqlite3_stmt *ptable_stmt = NULL, *pschema_stmt = NULL;
870  struct schema *schema = NULL;
871  int res, have_meta = 0;
872 
873  /* Get a list of table names */
874  if (sqlite3_prepare_v2(self->conn, table_stmt, -1, &ptable_stmt, NULL) != SQLITE_OK) {
875  logerror("sqlite:%s: Could not prepare statement '%s': %s\n",
876  database->name, table_stmt, sqlite3_errmsg(self->conn));
877  goto fail_exit;
878  }
879 
880  /* Check if the _experiment_metadata table exists */
881  do {
882  res = sqlite3_step(ptable_stmt);
883  if (res == SQLITE_ROW) {
884  if(strcmp ((const char*)sqlite3_column_text(ptable_stmt, 0), "_experiment_metadata") == 0) {
885  logdebug("sqlite:%s: Found table %s\n",
886  database->name, sqlite3_column_text(ptable_stmt, 0));
887  have_meta = 1;
888  }
889  }
890  } while (res == SQLITE_ROW && !have_meta);
891 
892  *num_tables = 0; /* In case !have_meta, we want it to be 0; also, we need it that way for later */
893 
894  if(!have_meta) {
895  logdebug("sqlite:%s: _experiment_metadata table not found\n", database->name);
896  /* XXX: This is probably a new database, don't exit in error */
897  sqlite3_finalize(ptable_stmt);
898  return NULL;
899  }
900 
901  /* Get schema for all tables */
902  if(sqlite3_reset(ptable_stmt) != SQLITE_OK) {
903  logwarn("sqlite:%s: Could not reset statement '%s' to traverse results again: %s\n",
904  database->name, table_stmt, sqlite3_errmsg(self->conn));
905  /* The _experiment_metadata table is often (always?) at the beginning of the list,
906  * so we try to recover gracefully from this issue by ignoring it */
907  }
908 
909  schema_stmt_ms = mstring_create();
910  do {
911  t = NULL;
912  mstring_set(schema_stmt_ms, "");
913 
914  res = sqlite3_step(ptable_stmt);
915  if (res == SQLITE_ROW) {
916  tablename = (const char*)sqlite3_column_text(ptable_stmt, 0);
917 
918  if(!strcmp (tablename, "_senders")) {
919  /* Create a phony entry for the _senders table some
920  * server/database.c:database_init() doesn't try to create it */
921  t = table_descr_new (tablename, NULL);
922 
923  } else {
924  mstring_sprintf(schema_stmt_ms, schema_stmt, tablename); /* XXX: Could a prepared statement do this concatenation? */
925  /* If it's *not* the _senders table, get its schema from the metadata table */
926  logdebug2("sqlite:%s:%s: Trying to find schema for table %s: %s\n",
927  database->name, __FUNCTION__, tablename,
928  mstring_buf(schema_stmt_ms));
929 
930  if (sqlite3_prepare_v2(self->conn, mstring_buf(schema_stmt_ms),
931  -1, &pschema_stmt, NULL) != SQLITE_OK) {
932  logerror("sqlite:%s: Could not prepare statement '%s': %s\n",
933  database->name, mstring_buf(schema_stmt_ms), sqlite3_errmsg(self->conn));
934  goto fail_exit;
935  }
936 
937  if(sqlite3_step(pschema_stmt) != SQLITE_ROW) {
938  logwarn("sqlite:%s: Could not get schema for table %s (or not present), ignoring it: %s\n",
939  database->name, tablename, sqlite3_errmsg(self->conn));
940 
941  } else {
942  meta = (const char*)sqlite3_column_text(pschema_stmt, 0); /* We should only have one result row */
943  schema = schema_from_meta(meta);
944  sqlite3_finalize(pschema_stmt);
945  pschema_stmt = NULL; /* Avoid double-finalisation in case of error */
946 
947  if (!schema) {
948  logerror("sqlite:%s: Could not parse schema '%s' (stored in DB) for table %s; is your database from an oml2-server<2.10?\n",
949  database->name, meta, tablename);
950  goto fail_exit;
951  }
952 
953  t = table_descr_new (tablename, schema);
954  if (!t) {
955  logerror("sqlite:%s: Could create table descrition for table %s\n",
956  database->name, tablename);
957  goto fail_exit;
958  }
959  schema = NULL; /* The pointer has been copied in t (see table_descr_new);
960  we don't want to free it twice in case of error */
961  }
962  }
963 
964  if (t) {
965  t->next = tables;
966  tables = t;
967  (*num_tables)++;
968  }
969  }
970  } while (res == SQLITE_ROW);
971 
972  mstring_delete(schema_stmt_ms);
973  sqlite3_finalize(pschema_stmt);
974  sqlite3_finalize(ptable_stmt);
975 
976  return tables;
977 
978 fail_exit:
979  if (tables) {
980  table_descr_list_free(tables);
981  }
982 
983  if (schema) {
984  schema_free(schema);
985  }
986  if (pschema_stmt) {
987  sqlite3_finalize(pschema_stmt);
988  }
989  if (schema_stmt_ms) {
990  mstring_delete(schema_stmt_ms);
991  }
992 
993  if (ptable_stmt) {
994  sqlite3_finalize(ptable_stmt);
995  }
996 
997  *num_tables = -1;
998  return NULL;
999 }
