98 static int
99 sample(
100  OmlFilter* f,
101  OmlValue* value
102 ) {
103  InstanceData* self = (InstanceData*)f->instance_data;
104  OmlValueU* v = oml_value_get_value(value);;
105  OmlValueT type = oml_value_get_type(value);;
106  double val;
107 
108  switch (type) {
109  case OML_LONG_VALUE:
110  val = v->longValue;
111  break;
112  case OML_DOUBLE_VALUE:
113  val = v->doubleValue;
114  break;
115  default:
116  // raise error;
117  return -1;
118  }
119  if (isnan(self->sample_sum)) {
120  self->sample_sum = val;
121  } else {
122  self->sample_sum += val;
123  }
124  if (val < self->sample_min || isnan(self->sample_min)) self->sample_min = val;
125  if (val > self->sample_max || isnan(self->sample_max)) self->sample_max = val;
126  self->sample_count++;
127  return 0;
128 }
