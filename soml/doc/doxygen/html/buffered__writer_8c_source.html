115  assert(chunkSize>=0);
116 
117  if((self = (BufferedWriter*)oml_malloc(sizeof(BufferedWriter)))) {
118  memset(self, 0, sizeof(BufferedWriter));
119 
120  self->outStream = outStream;
121  /* This forces a 'connected' INFO message upon first connection */
122  self->backoff = 1;
123 
124  self->bufSize = chunkSize > 0 ? chunkSize : DEF_CHAIN_BUFFER_SIZE;
125 
126  nchunks = queueCapacity / self->bufSize;
127  self->unallocatedBuffers = (nchunks > 2) ? nchunks : 2; /* at least two chunks */
128 
129  logdebug ("%s: Buffer size %dB (%d chunks of %dB)\n",
130  self->outStream->dest,
131  self->unallocatedBuffers*self->bufSize,
132  self->unallocatedBuffers, self->bufSize);
133 
134  if(NULL == (self->writerChunk = self->firstChunk = createBufferChunk(self))) {
135  oml_free(self);
136  self = NULL;
137 
138  } else if(NULL == (self->meta_buf = mbuf_create())) {
139  destroyBufferChain(self);
140  oml_free(self);
141  self = NULL;
142 
143  } else {
144  /* Initialize mutex and condition variable objects */
145  pthread_cond_init(&self->semaphore, NULL);
146  pthread_mutex_init(&self->lock, NULL);
147 
148  /* Initialize and set thread detached attribute */
149  pthread_attr_t tattr;
150  pthread_attr_init(&tattr);
151  pthread_attr_setdetachstate(&tattr, PTHREAD_CREATE_JOINABLE);

232 _bw_push(BufferedWriterHdl instance, uint8_t* data, size_t size)
233 {

235  if (!self->active) { return 0; }
236 
237  BufferChunk* chunk = self->writerChunk;
238  if (chunk == NULL) { return 0; }

413 BufferChunk*
414 getNextWriteChunk(BufferedWriter* self, BufferChunk* current) {
415  int nlost;
416  assert(current != NULL);
417  BufferChunk* nextBuffer = current->next;
418  assert(nextBuffer != NULL);
419 
420  if (mbuf_rd_remaining(nextBuffer->mbuf) == 0) {
421  // It's empty (the reader has finished with it), we can use it
422  mbuf_clear2(nextBuffer->mbuf, 0);
423  self->writerChunk = nextBuffer;
424  bw_msgcount_reset(self);
425 
426  } else if (self->unallocatedBuffers > 0) {
427  // Insert a new chunk between current and next one.
428  BufferChunk* newBuffer = createBufferChunk(self);
429  assert(newBuffer);
430  newBuffer->next = nextBuffer;
431  current->next = newBuffer;
432  self->writerChunk = newBuffer;
433 
434  } else {
435  // The chain is full, time to drop data and reuse the next buffer
436  current = nextBuffer;
437  assert(nextBuffer->reading == 0); /* Ensure this is not the chunk currently being read */
438  self->writerChunk = nextBuffer;
439 
440  nlost = bw_msgcount_reset(self);
441  self->nlost += nlost;
442  logwarn("Dropped %d samples (%dB)\n", nlost, mbuf_fill(nextBuffer->mbuf));
443  mbuf_repack_message2(self->writerChunk->mbuf);
444  }
445 
446  // Now we just need to copy the message from current to self->writerChunk
447  int msgSize = mbuf_message_length(current->mbuf);
448  if (msgSize > 0) {
449  mbuf_write(self->writerChunk->mbuf, mbuf_message(current->mbuf), msgSize);
450  mbuf_reset_write(current->mbuf);
451  bw_msgcount_add(self, 1);
452  }
453 
454  return self->writerChunk;
455 }

523 static void*
524 threadStart(void* handle)
525 {
526  int allsent = 1;
527  BufferedWriter* self = (BufferedWriter*)handle;
528  BufferChunk* chunk = self->firstChunk;
529 
530  while (self->active) {
531  oml_lock(&self->lock, "bufferedWriter");
532  pthread_cond_wait(&self->semaphore, &self->lock);
533  // Process all chunks which have data in them
534  do {
535  if (mbuf_message(chunk->mbuf) > mbuf_rdptr(chunk->mbuf)) {
536  allsent = processChunk(self, chunk);
537  }
538  // stop if we caught up to the writer
539 
540  if (chunk == self->writerChunk) break;
541 
542  if (allsent>0) {
543  chunk = chunk->next;
544  }
545  } while(allsent > 0);
546  oml_unlock(&self->lock, "bufferedWriter");
547  }
548  /* Drain this writer before terminating */
549  /* XXX: Backing-off for ... messages might confuse the user as
550  * we don't actually wait after a failure when draining at the end */
551  while ((allsent=processChunk(self, chunk))>=-1) {
552  if(allsent>0) {
553  if (chunk == self->writerChunk) break;
554  chunk = chunk->next;
555  } else if (-1 == allsent) {
556  sleep(self->backoff);
557  }
558  };
559  self->retval = allsent;
560  pthread_exit(&(self->retval));
561 }

571 static int
572 processChunk(BufferedWriter* self, BufferChunk* chunk)
573 {
574  assert(self);
575  assert(self->meta_buf);
576  assert(chunk);
577  assert(chunk->mbuf);
578 
579  uint8_t* buf = mbuf_rdptr(chunk->mbuf);
580  size_t size = mbuf_message_offset(chunk->mbuf) - mbuf_read_offset(chunk->mbuf);
581  size_t sent = 0;
582 
583  MBuffer* meta = self->meta_buf;
584 
585  /* XXX: Should we use a timer instead? */
586  time_t now;
587  time(&now);
588  if (difftime(now, self->last_failure_time) < self->backoff) {
589  logdebug("%s: Still in back-off period (%ds)\n", self->outStream->dest, self->backoff);
590  return -1;
591  }
592 
593  chunk->reading = 1;
594 
595  while (size > sent) {
596  long cnt = self->outStream->write(self->outStream, (void*)(buf + sent), size - sent,
597  meta->rdptr, meta->fill);
598  if (cnt > 0) {
599  sent += cnt;
600  if (self->backoff) {
601  self->backoff = 0;
602  loginfo("%s: Connected\n", self->outStream->dest);
603  }
604 
605  } else {
606  self->last_failure_time = now;
607  if (!self->backoff) {
608  self->backoff = 1;
609  } else if (self->backoff < UINT8_MAX) {
610  self->backoff *= 2;
611  }
612  logwarn("%s: Error sending, backing off for %ds\n", self->outStream->dest, self->backoff);
613 
614  chunk->reading = 0;
615  return -2;
616  }
617  }
618 
619  /* Check that we have sent everything, and reset the MBuffer
620  * XXX: is this really needed? size>sent *should* be enough
621  */
622  mbuf_read_skip(chunk->mbuf, sent);
623  chunk->reading = 0;
624  /* XXX: Redundant with allsent */
625  if (mbuf_write_offset(chunk->mbuf) == mbuf_read_offset(chunk->mbuf)) {
626  mbuf_clear2(chunk->mbuf, 1);
627  return 1;
628  }
629  return 0;
630 }
