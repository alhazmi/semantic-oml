222 int
223 parse_config(char* configFile)
224 {
225  xmlDocPtr doc;
226  xmlNodePtr cur;
227 
228  logdebug("Using configuration file '%s'\n", configFile);
229 
230  if ((doc = xmlParseFile(configFile)) == NULL) {
231  logerror("Config file '%s' not parsed successfully.\n",
232  configFile);
233  return -1;
234  }
235 
236  if ((cur = xmlDocGetRootElement(doc)) == NULL) {
237  logerror("Config file '%s' is empty.\n", configFile);
238  xmlFreeDoc(doc);
239  return -2;
240  }
241 
242  if (!match_xml_elt (cur, CT_ROOT)) {
243  logerror("Config file has incorrect root '%s', should be '%s'.\n",
244  cur->name, canonical_name (CT_ROOT));
245  xmlFreeDoc(doc);
246  return -3;
247  }
248 
249  if (omlc_instance->node_name == NULL) {
250  omlc_instance->node_name = get_xml_attr(cur, CT_NODE);
251  }
252  if (omlc_instance->domain == NULL) {
253  omlc_instance->domain = get_xml_attr(cur, CT_EXP);
254  }
255 
256  cur = cur->xmlChildrenNode;
257  while (cur != NULL) {
258  if (match_xml_elt(cur, CT_COLLECT)) {
259  if (parse_collector(cur)) {
260  xmlFreeDoc(doc);
261  return -4;
262  }
263  }
264  cur = cur->next;
265  }
266  xmlFreeDoc(doc);
267  return 0;
268 }

279 static int
280 parse_collector(xmlNodePtr el)
281 {
282  char* url = get_xml_attr(el, CT_COLLECT_URL);
283  OmlWriter* writer;
284  OmlMP *mp;
285  OmlMStream *ms;
286 
287  if (url == NULL) {
288  logerror("Config line %hu: Missing 'url' attribute for <%s ...>'.\n", el->line, el->name);
289  return -1;
290  }
291  char* encoding_s = get_xml_attr(el, CT_COLLECT_ENCODING);
292  enum StreamEncoding encoding;
293  if (encoding_s == NULL || strcmp(encoding_s, "binary") == 0) {
294  encoding = SE_Binary;
295  } else if (strcmp(encoding_s, "text") == 0) {
296  encoding = SE_Text;
297  } else {
298  logerror("Config line %hu: Unknown 'encoding' value '%s' for <%s ...>'.\n", el->line, encoding_s, el->name);
299  return -1;
300  }
301  if ((writer = create_writer(url, encoding)) == NULL) {
302  return -2;
303  }
304 
305  xmlNodePtr cur = el->xmlChildrenNode;
306  if (cur == NULL) {
307  logdebug("Found empty <collect /> element, sending all registered MPs to %s\n", url);
308  mp = omlc_instance->mpoints;
309  if (omlc_instance->sample_interval == 0 &&
310  omlc_instance->sample_count == 0) {
311  /* Sampling or intervals work on a per <stream /> basis as we are
312  * handling the case where they are not specifically defined here by
313  * putting a sane default. See #541 for some discussion about this */
314  omlc_instance->sample_count = 1;
315  }
316  while (mp != NULL) {
317  ms = oml_mp_get_default_ms(mp);
318  if(!ms) {
319  logerror("Error creating MS %s towards %s, skipping...\n", mp->name, url);
320  } else {
321  oml_ms_add_writer(ms, writer);
322  }
323  mp = mp->next;
324  }
325  } else {
326  /* Unconditionally add metadata stream,
327  * as it otherwise confuses the server */
328  if(add_metadata_stream(writer)) {
329  return -3;
330  }
331 
332  while (cur != NULL) {
333  if (match_xml_elt (cur, CT_STREAM)) {
334  if (parse_stream_or_mp(cur, writer)) {
335  return -3;
336  }
337  }
338  cur = cur->next;
339  }
340  }
341 
342  return 0;
343 }

380 static int
381 parse_stream_or_mp (xmlNodePtr el, OmlWriter* writer)
382 {
383  /*
384  * CT_STREAM_SOURCE is a special case because we mix the old names
385  * with the new: the 'name' attribute identifies the source MP in
386  * the old <mp ...> naming, but it identifies the created _stream_
387  * in the new naming. It requires special handling as a result.
388  */

390  return parse_mp (el, writer);
391  else

445 static int
446 parse_stream_filters (xmlNodePtr el, OmlWriter *writer,
447  char *source, char *name)
448 {
449  OmlMP* mp;
450  char *samples_str = get_xml_attr (el, CT_STREAM_SAMPLES);
451  char *interval_str = get_xml_attr (el, CT_STREAM_INTERVAL);
452 
453  if (source == NULL) {
454  logerror("Config line %hu: Missing 'name' attribute for <%s ...>'.\n",
455  el->line, el->name);
456  return -1;
457  }
458 
459  mp = find_mp (source);
460 
461  if (mp == NULL) {
462  logerror("Config line %hu: Unknown measurement point '%s'.\n", el->line, source);
463  return -4;
464  }
465 
466  if (samples_str == NULL && interval_str == NULL) {
467  logerror("Config line %hu: Missing 'samples' or 'interval' attribute for <%s ...>'.\n",
468  el->line, el->name);
469  return -2;
470  }
471  if (samples_str != NULL && interval_str != NULL) {
472  logerror("Config line %hu: Only one of 'samples' or 'interval' attribute can"
473  " be defined for <%s ...>.\n",
474  el->line,
475  el->name);
476  return -3;
477  }
478 
479  int samples = samples_str != NULL ? atoi((char*)samples_str) : -1;
480  if (samples == 0) samples = 1;
481  double interval = interval_str != NULL ? atof((char*)interval_str) : -1;
482  OmlMStream* ms = create_mstream(name, mp, writer, interval, samples);
483 
484  if (!ms)
485  return -6;
486 
487  xmlNodePtr el2 = el->children;
488  for (; el2 != NULL; el2 = el2->next) {
489  if (match_xml_elt (el2, CT_FILTER)) {
490  OmlFilter* f = parse_filter(el2, ms, mp);
491  if (f == NULL) {
492  // too difficult to reclaim all memory
493  return -5;
494  }
495  f->next = ms->filters;
496  ms->filters = f;
497  }
498  }
499 
500  // no filters specified - use default
501  if (ms->filters == NULL) {
502  create_default_filters(mp, ms);
503  }
504  if (interval > 0) {
505  filter_engine_start(ms);
506  }
507  return 0;
508 }

517 static OmlFilter*
518 parse_filter (xmlNodePtr el, OmlMStream* ms, OmlMP* mp)
519 {
520  OmlFilter *f = NULL;
521  OmlMPDef *def = NULL;
522  char* field = get_xml_attr (el, CT_FILTER_FIELD);
523  char* operation = get_xml_attr (el, CT_FILTER_OPER);
524  char* rename = get_xml_attr (el, CT_FILTER_RENAME);
525  int index = -1;
526 
527  if (field == NULL) {
528  logerror("Config line %hu: Filter config element <%s ...> must include a '%s' attribute.\n",
529  el->line, el->name, canonical_name (CT_FILTER_FIELD));
530  return NULL;
531  }
532 
533  index = find_mp_field (field, mp);
534 
535  /* If index == -1, we didn't find the named field. This is an error, so we should abort. */
536  if (index == -1) {
537  MString *s = mp_fields_summary (mp);
538  logerror ("Config line %hu: MP '%s' has no field named '%s'; "
539  "Valid fields for '%s' are: %s\n",
540  el->line, mp->name, field, mp->name, mstring_buf (s));
541  mstring_delete (s);
542  return NULL;
543  }
544 
545  def = &mp->param_defs[index];
546  if (operation == NULL) {
547  // pick default one
548  if (def == NULL) {
549  logerror("Config line %hu: Can't create default filter without '%s' declaration.\n",
550  el->line, canonical_name (CT_FILTER_FIELD));
551  return NULL;
552  }
553  f = create_default_filter(def, ms, index);
554  } else {
555  const char* name = (rename != NULL) ? (char*)rename : def->name;
556  f = create_filter((const char*)operation, name, def->param_types, def->relations, index);
557  }
558  if (f != NULL) {
559  f = parse_filter_properties(el, f);
560  }
561  return f;
562 }

579 static OmlFilter*
580 parse_filter_properties(xmlNodePtr el, OmlFilter* f)
581 {
582  xmlNodePtr el2 = el->children;
583  for (; el2 != NULL; el2 = el2->next) {
584  if (match_xml_elt (el2, CT_FILTER_PROP)) {
585  if (f->set == NULL) {
586  char *operation = get_xml_attr (el, CT_FILTER_OPER);
587  logerror("Filter '%s' doesn't support setting properties.\n",
588  operation);
589  return NULL;
590  }
591 
592  char* pname = get_xml_attr (el2, CT_FILTER_PROP_NAME);
593  if (pname == NULL) {
594  logerror("Config line %hu: Filter property declared without a name in filter '%s'\n",
595  el2->line, f->name);
596  return NULL;
597  }
598  char* ptype = get_xml_attr (el2, CT_FILTER_PROP_TYPE);
599  if (ptype == NULL) ptype = "string";
600 
601  xmlNodePtr vel = el2->children;
602  xmlChar* value = NULL;
603  for (; vel != NULL; vel = vel->next) {
604  if (vel->type == XML_TEXT_NODE) {
605  value = vel->content;
606  break;
607  }
608  }
609  if (value == NULL) {
610  logerror("Config line %hu: Missing value for property '%s' in filter '%s'.\n",
611  el2->line, pname, f->name);
612  return NULL;
613  }
614  logdebug("Found filter property: %s:%s = '%s'.\n",
615  pname, ptype, value);
616  set_filter_property(f, (char*)pname, (char*)ptype, (char*)value);
617  }
618  }
619  return f;
620 }
