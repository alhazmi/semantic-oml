25 mbuf_check_invariant (MBuffer* mbuf)
26 {
27  assert (mbuf != NULL);
28  assert (mbuf->base != NULL);
29  assert (mbuf->wr_remaining <= mbuf->length);

32  assert (mbuf->wrptr >= mbuf->rdptr);
33  assert (mbuf->wrptr - mbuf->base == (int)mbuf->fill);
34  assert ((int)mbuf->rd_remaining == (mbuf->wrptr - mbuf->rdptr));
35  assert ((mbuf->msgptr <= mbuf->rdptr) || (mbuf->msgptr <= mbuf->wrptr));

302 int
303 mbuf_resize (MBuffer* mbuf, size_t new_length)
304 {
305  mbuf_check_invariant (mbuf);
306 
307  if (mbuf == NULL || mbuf->base == NULL)
308  return -1;
309 
310  /* Don't resize if we don't have to */
311  if (new_length <= mbuf->length)
312  return 0;
313 
314  if (! mbuf->allow_resizing)
315  return -1;
316 
317 /* // Make sure we grow at leat by a minimum delta */
318 /* if ((new_length - mbuf->length) < mbuf->min_resize) { */
319 /* new_length = mbuf->length + mbuf->min_resize; */
320 /* } */
321 
322  int wr_offset = mbuf->wrptr - mbuf->base;
323  int rd_offset = mbuf->rdptr - mbuf->base;
324  int msg_offset = mbuf->msgptr - mbuf->base;
325 
326  assert (wr_offset == (int)mbuf->fill);
327 
328  uint8_t* new = oml_realloc (mbuf->base, new_length);
329  if (new == NULL)
330  return -1;
331 
332  mbuf->base = new;
333  mbuf->wrptr = mbuf->base + wr_offset;
334  mbuf->rdptr = mbuf->base + rd_offset;
335  mbuf->msgptr = mbuf->base + msg_offset;
336 
337  mbuf->resized += (new_length - mbuf->length);
338  mbuf->length = new_length;
339  mbuf->wr_remaining = mbuf->length - mbuf->fill;
340 
341  mbuf_check_invariant (mbuf);
342 
343  return 0;
344 }

353 int
354 mbuf_check_resize (MBuffer* mbuf, size_t bytes)
355 {
356  if (mbuf->wr_remaining < bytes) {
357  size_t inc = bytes - mbuf->wr_remaining;
358  size_t min_resize = mbuf->min_resize > 0 ? mbuf->min_resize : mbuf->length;
359  inc = min_resize * (inc / min_resize + 1);
360  return mbuf_resize (mbuf, mbuf->length + inc);
361  } else {
362  return 0;
