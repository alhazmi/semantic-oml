213 Database*
214 database_find (const char* name)
215 {
216  Database* db = first_db;
217  while (db != NULL) {
218  if (!strcmp(name, db->name)) {
219  loginfo ("%s: Database already open (%d client%s)\n",
220  name, db->ref_count, db->ref_count>1?"s":"");
221  db->ref_count++;
222  return db;
223  }
224  db = db->next;
225  }
226 
227  // need to create a new one
228  Database *self = oml_malloc(sizeof(Database));
229  logdebug("%s: Creating or opening database\n", name);
230  strncpy(self->name, name, MAX_DB_NAME_SIZE);
231  self->ref_count = 1;
232  self->create = database_create_function (dbbackend);
233 
234  if (self->create (self)) {
235  oml_free(self);
236  return NULL;
237  }
238 
239  if (database_init (self) == -1) {
240  database_release(self);
241  oml_free (self);
242  return NULL;
243  }
244 
245  char *start_time_str = self->get_metadata (self, "start_time");
246 
247  if (start_time_str == NULL) {
248  /* No start time: this is probably a new database */
249  database_hook_send_event(self, HOOK_CMD_DBCREATED);
250 
251  } else {
252  database_hook_send_event(self, HOOK_CMD_DBOPENED);
253  self->start_time = strtol (start_time_str, NULL, 0);
254  oml_free (start_time_str);
255  logdebug("%s: Retrieved start-time = %lu\n", name, self->start_time);
256  }
257 
258  // hook this one into the list of active databases
259  self->next = first_db;
260  first_db = self;
261 
262  return self;
263 }
269 void
270 database_release(Database* self)
271 {
272  if (self == NULL) {
273  logerror("NONE: Trying to release a NULL database.\n");
274  return;
275  }
276  if (--self->ref_count > 0) return; // still in use
277 
278  // unlink DB
279  Database* db_p = first_db;
280  Database* prev_p = NULL;
281  while (db_p != NULL && db_p != self) {
282  prev_p = db_p;
283  db_p = db_p->next;
284  }
285  if (db_p == NULL) {
286  logerror("%s: Trying to release an unknown database\n", self->name);
287  return;
288  }
289  if (prev_p == NULL)
290  first_db = self->next; // was first
291  else
292  prev_p->next = self->next;
293 
294  // no longer needed
295  DbTable* t_p = self->first_table;
296  while (t_p != NULL) {
297  DbTable* t = t_p->next;
298  /* Release the backend storage for this table */
299  self->table_free (self, t_p);
300  /* Release the table */
301  database_table_free(self, t_p);
302  t_p = t;
303  }
304 
305  loginfo ("%s: Closing database\n", self->name);
306  self->release (self);
307 
308  database_hook_send_event(self, HOOK_CMD_DBCLOSED);
309 
310  oml_free(self);
311 }

395 DbTable*
396 database_find_or_create_table(Database *database, struct schema *schema)
397 {
398  if (database == NULL) return NULL;
399  if (schema == NULL) return NULL;
400 
401  DbTable *table = NULL;
402  struct schema *s = schema_copy(schema);
403  int i = 1;
404  int diff = 0, tnlen;
405 
406  tnlen = strlen(schema->name);
407 
408  do {
409  table = database_find_table (database, s->name);
410 
411  if (table) {
412  diff = schema_diff (s, table->schema);
413  if (!diff) {
414  schema_free(s);
415  return table;
416 
417  } else if (diff == -1) {
418  logerror ("%s: Schema error table '%s'\n", database->name, s->name);
419  logdebug (" One of the server schema %p or the client schema %p is probably NULL\n", s, table->schema);
420 
421  } else if (diff > 0) {
422  logdebug ("%s: Schema differ for table '%s', at or after column %d\n",
423  database->name, s->name, diff);
424  }
425 
426  if (i == 1) {
427  /* First time we need to increase the size */
428  /* Add space for 2 characters and null byte, that is up to '_9', with MAX_TABLE_RENAME = 10 */
429  s->name = oml_realloc(s->name, tnlen + 3);
430  strncpy(s->name, schema->name, tnlen);
431  }
432  snprintf(&s->name[tnlen], 3, "_%d", ++i);
433  }
434 
435  } while(table && diff && (i < MAX_TABLE_RENAME));
436 
437  if (table && diff) {
438  logerror ("%s: Too many (>%d) tables named '%s_x', giving up. Please use the rename attribute of <mp /> tags.\n",
439  database->name, MAX_TABLE_RENAME, schema->name);
440  schema_free(s);
441  return NULL;
442  }
443 
444  if(i>1) {
445  /* We had to change the table name*/
446  logwarn("%s: Creating table '%s' for new stream '%s' with incompatible schema\n", database->name, s->name, schema->name);
447  oml_free(schema->name);
448  schema->name = oml_strndup(s->name, tnlen+3);
449  }
450  schema_free(s);
451 
452  /* No table by that name exists, so we create it */
453  table = database_create_table (database, schema);
454  if (!table)
455  return NULL;
456  if (database->table_create (database, table, 0)) {
457  logerror ("%s: Couldn't create table '%s'\n", database->name, schema->name);
458  /* Unlink the table from the experiment's list */
459  DbTable* t = database->first_table;
460  if (t == table)
461  database->first_table = t->next;
462  else {
463  while (t && t->next != table)
464  t = t->next;
465  if (t && t->next)
466  t->next = t->next->next;
467  }
468  database_table_free (database, table);
469  return NULL;
470  }
471  return table;
472 }

516 int
517 database_init (Database *database)
518 {
519  int num_tables;
520  TableDescr* tables = database->get_table_list (database, &num_tables);
521  if (!database->semantic)
522  {
523  TableDescr* td = tables;
524 
525  if (num_tables == -1)
526  return -1;
527 
528  logdebug("%s: Got table list with %d tables in it\n", database->name, num_tables);
529  int i = 0;
530  for (i = 0; i < num_tables; i++, td = td->next) {
531  if (td->schema) {
532  struct schema *schema = schema_copy (td->schema);
533  DbTable *table = database_create_table (database, schema);
534 
535  if (!table) {
536  logwarn ("%s: Failed to create table '%s'\n",
537  database->name, td->name);
538  continue;
539  }
540  /* Create the required table data structures, but don't do SQL CREATE TABLE */
541  if (database->table_create (database, table, 1) == -1) {
542  logwarn ("%s: Failed to create adapter structures for table '%s'\n",
543  database->name, td->name);
544  database_table_free (database, table);
545  }
546  }
547  }
548 
549  /* Create default tables if they are not already present in the 'tables' list */
550  const char *meta_tables [] = { "_senders", "_experiment_metadata" };
551  size_t j;
552  for (j = 0; j < LENGTH(meta_tables); j++) {
553  if (!table_descr_have_table (tables, meta_tables[j])) {
554  if (database->table_create_meta (database, meta_tables[j])) {
555  table_descr_list_free (tables);
556  logerror("%s: Could not create default table %s\n",
557  database->name, meta_tables[j]);
558  return -1;
559  }
560  }
561  }
562 
563  table_descr_list_free (tables);
564  }
565  return 0;
566 }
