208 enum HeaderTag
209 tag_from_string (const char *str, size_t n)
210 {
211  int i = 0;
212  char first;
213  char second;
214 
215  if (str == NULL || n <= 2)
216  return H_NONE;
217 
218  first = *str;
219  second = *(str+1);
220 
221  /* Premature optimization for fun and profit */
222  for (; header_map[i].name != NULL; i++) {
223  char h_first = header_map[i].name[0];
224  char h_second = header_map[i].name[1];
225  size_t h_len = header_map[i].namelen;
226 
227  if (n < h_len)
228  continue;
229 
230  if (first != 's') {
231  /* We can fail on mismatched first char if it's not 's' */
232  if ((first == h_first) &&
233  (strncmp (str+1, header_map[i].name+1, n - 1) == 0))
234  return header_map[i].tag;
235  } else {
236  /* If 's', we can fast-fail on the second char */
237  if (first == h_first && second == h_second &&
238  (strncmp (str+2, header_map[i].name+2, n - 2) == 0)) {
239  return header_map[i].tag;
240  }
241  }
242  }
243  return H_NONE;
244 }

286 header_from_string (const char *str, size_t n)
287 {
288  if (!str) return NULL;
289 
290  const char *p = str;
291  const char *q = memchr (p, ':', n);
292  if (!q)
293  return NULL;
294 
295  /* Strip trailing whitespace on the header name */
296  p = find_white(p);
297 
298  int namelen;
299  if (p < q)
300  namelen = p - str;
301  else
302  namelen = q - str;
303 
304  enum HeaderTag tag = tag_from_string (str, namelen);
305  if (tag == H_NONE)
306  return NULL;
307 
308  /* Skip the ':' and strip leading whitespace on header value */
309  if (*q)
310  q = skip_white (++q);
311 
312  char * value = NULL;
313  size_t valuelen = n - (q - str);
314  struct header *header = NULL;
315 
316  if ((int)valuelen > 0) {
317  value = oml_strndup (q, valuelen);
318  if (!value)
319  return NULL;
320 
321  header = oml_malloc (sizeof (struct header));
322  if (!header) {
323  oml_free (value);
324  return NULL;
325  }
326  header->tag = tag;
327  header->value = value;
328  return header;
329  }
330 
331  return NULL;
332 }
