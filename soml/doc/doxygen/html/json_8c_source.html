37 static ssize_t
38 resize_buffer(char **str, size_t new_sz)
39 {
40  assert(str);
41  ssize_t str_sz = -1;
42  if(NULL == *str || oml_malloc_usable_size(*str) < new_sz) {
43  char *new = oml_realloc(*str, new_sz);
44  if(new) {
45  *str = new;
46  str_sz = new_sz;
47  memset(*str, 0, str_sz);
48  }
49  } else {
50  str_sz = oml_malloc_usable_size(*str);
51  memset(*str, 0, str_sz);
52  }
53  return str_sz;

74  o_log(O_LOG_ERROR, "%s(): failed to resize buffer for [double] (v_sz=%zu)\n", __func__, v_sz);
75  return -1;
76  }
77
