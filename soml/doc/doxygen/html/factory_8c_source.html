91 create_filter(
92  const char* filter_type,
93  const char* instance_name,
94  OmlValueT type,
95  OMLSemDef *concepts,
96  int index
97 ) {
98  FilterType* ft = filter_types;
99  for (; ft != NULL; ft = ft->next) {
100  if (strcmp (filter_type, ft->name) == 0) break;
101  }
102  if (ft == NULL) {
103  logerror ("Unknown filter '%s'.\n", filter_type);
104  return NULL; // nothing found
105  }
106 
107  OmlFilter * f = (OmlFilter*)oml_malloc(sizeof(OmlFilter));
108  memset(f, 0, sizeof(OmlFilter));
109 
110  strncpy (f->name, instance_name, sizeof(f->name));
111  if (f->name[sizeof(f->name)-1] != '\0')
112  f->name[sizeof(f->name)-1] = '\0';
113  f->index = index;
114  f->concepts = concepts;
115  f->set = ft->set;
116  f->input_type = type;
117  f->input = ft->input;
118  f->output = ft->output;
119  f->newwindow = ft->newwindow;
120  f->meta = ft->meta;
121  f->definition = ft->definition; /* FIXME: Copy and substitute OML_INPUT_VALUE types */
122  f->output_count = ft->output_count;
123  f->result = create_filter_result_vector (f->definition, type, ft->output_count);
124  f->instance_data = ft->create(type, f->result);
125 
126  return f;

209 int
210 omlf_register_filter(const char* filter_name,
211  oml_filter_create create,
212  oml_filter_set set,
213  oml_filter_input input,
214  oml_filter_output output,
215  oml_filter_newwindow newwindow,
216  oml_filter_meta meta,
217  OmlFilterDef* filter_def)
218 {
219  if (create == NULL || input == NULL || output == NULL) {
220  logerror ("Filter %s needs a create function, an input function, and an output function (one of them was null).\n", filter_name);
221  return -1;
222  }
223 
224  if (filter_def == NULL) {
225  logerror ("Filter %s needs a filter definition (got a null definition).\n", filter_name);
226  return -1;
227  }
228 
229  FilterType* ft = (FilterType*)oml_malloc(sizeof(FilterType));
230  ft->name = filter_name;
231  ft->create = create;
232  ft->set = set;
233  ft->input = input;
234  ft->output = output;
235  ft->newwindow = newwindow;
236  ft->output_count = 0;
237 
238  OmlFilterDef* dp = filter_def;
239  for (; !(dp->name == NULL && dp->type == 0); dp++)
240  ft->output_count++;
241 
242  ft->definition = copy_filter_definition (filter_def, ft->output_count);
243 
244  if (set == NULL)
245  ft->set = default_filter_set;
246 
247  if (meta == NULL)
248  ft->meta = default_filter_meta;
249  else
250  ft->meta = meta;
251 
252  ft->next = filter_types;
253  filter_types = ft;
254 
255  return 0;
256 }
