75 int
76 read_message (char **line, size_t *size, size_t *length_out, int *seqno_out)
77 {
78  int length = 0;
79  int seqno = 0; // Message sequence number
80  int count = 0;
81  int result = 0;
82  char header[HEADER_LENGTH+1] = {0};
83 
84  bzero (*line, *size);
85 
86  do {
87  result = read (0, header + count, HEADER_LENGTH - count);
88  if (result == -1) {
89  fprintf (stderr, "reading header: %s\n", strerror (errno));
90  return -1;
91  } else if (result == 0) {
92  fprintf (stderr, "\n# msgloop: no more data, assuming closed pipe\n");
93  return 0;
94  }
95 
96  count += result;
97  } while (count < HEADER_LENGTH);
98  count = 0;
99 
100  if (result < HEADER_LENGTH) {
101  fprintf (stderr, "Couldn't read the whole header length at once!\n");
102  fprintf (stderr, "%d: %s\n", result, header);
103  return -1;
104  }
105 
106  if (strncmp ("OML", header, 3) != 0) {
107  fprintf (stderr, "Packet format error (not OML)\n");
108  return -1;
109  }
110 
111  seqno = read_ascii_hex_int (&header[3], 8);
112  length = read_ascii_hex_int (&header[11], 8);
113 
114  if (*size < length) {
115  char *new = realloc (*line, length);
116  if (new == NULL) {
117  fprintf (stderr, "realloc() error! %s\n", strerror (errno));
118  return -1;
119  }
120  *line = new;
121  *size = length;
122  }
123 
124  char *p = *line;
125  do {
126  result = read (0, p + count, length - count);
127  if (result == -1) {
128  fprintf (stderr, "Error reading message payload: %s\n", strerror (errno));
129  return -1;
130  } else if (result == 0) {
131  fprintf (stderr, "Read no bytes! Pipe closed?\n");
132  return 0;
133  }
134 
135  count += result;
136  } while (count < length);
137 
138  *length_out = length;
139  *seqno_out = seqno;
140  return 1;
141 }
