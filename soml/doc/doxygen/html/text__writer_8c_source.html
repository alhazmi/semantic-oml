139 static int
140 owt_row_cols(OmlWriter* writer, OmlValue* values, int value_count)
141 {
142  char *enc;
143  OmlTextWriter* self = (OmlTextWriter*)writer;
144  MBuffer* mbuf;
145  if ((mbuf = self->mbuf) == NULL) {
146  return 0; /* previous use of mbuf failed */
147  }
148 
149  int i;
150  OmlValue* v = values;
151  for (i = 0; i < value_count; i++, v++) {
152  int res;
153  switch (oml_value_get_type(v)) {
154  case OML_LONG_VALUE:
155  res = mbuf_print(mbuf, "\t%" PRId32, oml_value_clamp_long (omlc_get_long(*oml_value_get_value(v))));
156  break;
157 
158  case OML_INT32_VALUE:
159  res = mbuf_print(mbuf, "\t%" PRId32,
160  omlc_get_int32(*oml_value_get_value(v)));
161  break;
162  case OML_UINT32_VALUE:
163  res = mbuf_print(mbuf, "\t%" PRIu32,
164  omlc_get_uint32(*oml_value_get_value(v)));
165  break;
166  case OML_INT64_VALUE:
167  res = mbuf_print(mbuf, "\t%" PRId64,
168  omlc_get_int64(*oml_value_get_value(v)));
169  break;
170  case OML_UINT64_VALUE:
171  res = mbuf_print(mbuf, "\t%" PRIu64,
172  omlc_get_uint64(*oml_value_get_value(v)));
173  break;
174 
175  case OML_DOUBLE_VALUE:
176  res = mbuf_print(mbuf, "\t%f",
177  omlc_get_double(*oml_value_get_value(v)));
178  break;
179 
180  case OML_STRING_VALUE:
181  if(omlc_get_string_ptr(*oml_value_get_value(v)) &&
182  0 < omlc_get_string_length(*oml_value_get_value(v))) {
183  enc = oml_malloc(backslash_encode_size(omlc_get_string_size(v->value)));
184  backslash_encode(omlc_get_string_ptr(v->value), enc);
185  res = mbuf_print(mbuf, "\t%s", enc);
186  oml_free(enc);
187 
188  } else {
189  logdebug ("Attempting to send NULL or empty string; string of length 0 will be sent\n");
190  res = mbuf_print(mbuf, "\t");
191  }
192  break;
193 
194  case OML_BLOB_VALUE: {
195  if(omlc_get_blob_ptr(*oml_value_get_value(v)) &&
196  0 < omlc_get_blob_length(*oml_value_get_value(v))) {
197  enc = oml_malloc(base64_size_string(omlc_get_blob_length(*oml_value_get_value(v))));
198  base64_encode_blob(omlc_get_blob_length(*oml_value_get_value(v)), omlc_get_blob_ptr(*oml_value_get_value(v)), enc);
199  res = mbuf_print(mbuf, "\t%s", enc);
200  oml_free(enc);
201 
202  } else {
203  logdebug ("Attempting to send NULL or empty blob; blob of length 0 will be sent\n");
204  res = mbuf_print(mbuf, "\t");
205  }
206  break;
207  }
208 
209  case OML_GUID_VALUE:
210  res = mbuf_print(mbuf, "\t%" PRIu64, omlc_get_guid(*oml_value_get_value(v)));
211  break;
212 
213  case OML_BOOL_VALUE:
214  res = mbuf_print(mbuf, "\t%c", (omlc_get_bool(*oml_value_get_value(v))!=OMLC_BOOL_FALSE)?'T':'F');
215  break;
216 
217  case OML_VECTOR_DOUBLE_VALUE: {
218  OmlValueU *u = oml_value_get_value(v);
219  size_t nof_elts = omlc_get_vector_nof_elts(*u);
220  double *elts = omlc_get_vector_ptr(*u);
221  res = mbuf_print(mbuf, "\t%zu", nof_elts);
222  for(i = 0; 0 == res && i < nof_elts; i++)
223  res = mbuf_print(mbuf, " %.*g", DBL_DIG, elts[i]);
224  break;
225  }
226 
227  case OML_VECTOR_INT32_VALUE: {
228  OmlValueU *u = oml_value_get_value(v);
229  size_t nof_elts = omlc_get_vector_nof_elts(*u);
230  int32_t *elts = omlc_get_vector_ptr(*u);
231  res = mbuf_print(mbuf, "\t%zu", nof_elts);
232  for(i = 0; 0 == res && i < nof_elts; i++)
233  res = mbuf_print(mbuf, " %" PRId32, elts[i]);
234  break;
235  }
236 
237  case OML_VECTOR_UINT32_VALUE: {
238  OmlValueU *u = oml_value_get_value(v);
239  size_t nof_elts = omlc_get_vector_nof_elts(*u);
240  uint32_t *elts = omlc_get_vector_ptr(*u);
241  res = mbuf_print(mbuf, "\t%zu", nof_elts);
242  for(i = 0; 0 == res && i < nof_elts; i++)
243  res = mbuf_print(mbuf, " %" PRIu32, elts[i]);
244  break;
245  }
246 
247  case OML_VECTOR_INT64_VALUE: {
248  OmlValueU *u = oml_value_get_value(v);
249  size_t nof_elts = omlc_get_vector_nof_elts(*u);
250  int64_t *elts = omlc_get_vector_ptr(*u);
251  res = mbuf_print(mbuf, "\t%zu", nof_elts);
252  for(i = 0; 0 == res && i < nof_elts; i++)
253  res = mbuf_print(mbuf, " %" PRId64, elts[i]);
254  break;
255  }
256 
257  case OML_VECTOR_UINT64_VALUE: {
258  OmlValueU *u = oml_value_get_value(v);
259  size_t nof_elts = omlc_get_vector_nof_elts(*u);
260  uint64_t *elts = omlc_get_vector_ptr(*u);
261  res = mbuf_print(mbuf, "\t%zu", nof_elts);
262  for(i = 0; 0 == res && i < nof_elts; i++)
263  res = mbuf_print(mbuf, " %" PRIu64, elts[i]);
264  break;
265  }
266 
267  case OML_VECTOR_BOOL_VALUE: {
268  OmlValueU *u = oml_value_get_value(v);
269  size_t nof_elts = omlc_get_vector_nof_elts(*u);
270  bool *elts = omlc_get_vector_ptr(*u);
271  res = mbuf_print(mbuf, "\t%zu", nof_elts);
272  for(i = 0; 0 == res && i < nof_elts; i++)
273  res = mbuf_print(mbuf, " %s", elts[i] ? "True" : "False");
274  break;
275  }
276 
277  default:
278  res = -1;
279  logerror("%s: Unsupported value type '%d'\n", __FUNCTION__, oml_value_get_type(v));
280  return 0;
281  }
282 
283  if (res < 0) {
284  mbuf_reset_write(mbuf);
285  self->mbuf = NULL;
286  return 0;
287  }
288  }
289  return 1;
290 }

295 static int
296 owt_row_start(OmlWriter* writer, OmlMStream* ms, double now)
297 {
298  OmlTextWriter* self = (OmlTextWriter*)writer;
299  assert(self->bufferedWriter != NULL);
300 
301  MBuffer* mbuf;
302  if ((mbuf = self->mbuf = bw_get_write_buf(self->bufferedWriter, 1)) == NULL) {
303  return 0;
304  }
305 
306  mbuf_begin_write(mbuf);
307  if (mbuf_print(mbuf, "%f\t%d\t%ld", now, ms->index, ms->seq_no)) {
308  mbuf_reset_write(mbuf);
309  self->mbuf = NULL;
310  return 0;
311  }
312  return 1;
313 }

318 static int
319 owt_row_end(OmlWriter* writer, OmlMStream* ms)
320 {
321  (void)ms;
322  OmlTextWriter* self = (OmlTextWriter*)writer;
323  MBuffer* mbuf;
324  if ((mbuf = self->mbuf) == NULL) {
325  return 0; /* previous use of mbuf failed */
326  }
327 
328  int res;
329  if ((res = mbuf_write(mbuf, (uint8_t*)"\n", 1)) != 0) {
330  mbuf_reset_write(mbuf);
331 
332  } else {
333  // success, lock in message
334  if (0 == ms->index) {
335  /* This is schema0, also push the data into the meta_buf
336  * to be replayed after a disconnection.
337  *
338  * At the moment, the oml_outs_write_f takes header information as a
339  * whole, but does not push more once it has sent the initial block. Its
340  * two last parameters are only used to resend the entirety of the headers
341  * when a disconnection does occur, nothing before.
342  *
343  * We therefore send the extra piece of data the normal way, but also
344  * record it, separately, in the meta_buf
345  *
346  * XXX: This logic should be in higher layer levels, but given the current
347  * implementation, with some of it already spread down into the
348  * OmlOutStream (oml_outs_write_f), this require a much bigger refactoring.
349  * It is also duplicated with the OmlBinWriter (see #1101).
350  */
351  _bw_push_meta(self->bufferedWriter,
352  mbuf_message(self->mbuf), mbuf_message_length(self->mbuf));
353  }
354 
355  mbuf_begin_write (mbuf);
356  }
357 
358  self->mbuf = NULL;
359  bw_msgcount_add(self->bufferedWriter, 1);
360  bw_unlock_buf(self->bufferedWriter);
361  return res == 0;
362 }
