49 static void xcount_new (size_t bytes) {
50 #if OML_MEM_DEBUG
51  o_log(O_LOG_DEBUG4, "Allocated %dB of memory\n", bytes);
52 #endif

89 oml_memsummary_r (char *summary, size_t summary_sz)
90 {
91  size_t xbytes_h = xbytes;
92  char *units = "bytes";
93  if (xbytes_h > 10*(1<<10)) {
94  units = "KiB";
95  xbytes_h >>= 10;

98  units = "MiB";
99  xbytes_h >>= 10;

102  snprintf(summary, summary_sz, "%"PRIuMAX" %s currently allocated [%"
103  PRIuMAX" allocated overall, %"
104  PRIuMAX" freed, %"
105  PRIuMAX" current, %"
106  PRIuMAX" maximum]",
107  (uintmax_t)xbytes_h, units,

157 oml_malloc (size_t size)
158 {
159  size += sizeof (size_t);
160  void *ret = malloc (size);
161  if (!ret)
162  xreturn (ret, size, "Out of memory, malloc failed\n");
163  memset (ret, 0, size);

176 void*
177 oml_calloc (size_t count, size_t size)
178 {
179  if (size >= sizeof (size_t))
180  count++;
181  else {
182  int n = 0;
183  size_t ssize = sizeof (size_t);
184  while (ssize > size) {
185  ssize >>= 1;
186  n++;
187  }
188  count += (1 << n);
189  }
190  void *ret = calloc (count, size);
191  if (!ret)
192  xreturn (ret, count * size, "Out of memory, calloc failed\n");
193  *(size_t*)ret = count * size;
194  xcount_new (count * size);
195  return (size_t*)ret + 1;
196 }

205 void*
206 oml_realloc (void *ptr, size_t size)
207 {
208  if (!ptr) return oml_malloc (size);
209  size += sizeof (size_t);
210  ptr = (size_t*)ptr - 1;
211  size_t old = *(size_t*)ptr;
212  void *ret = realloc (ptr, size);
213  if (!ret)
214  xreturn (ret, size - old, "Out of memory, realloc failed\n");
215  *(size_t*)ret = size;
216  xcount_new (size);
217  xcount_freed (old);
218  return (size_t*)ret + 1;
219 }
