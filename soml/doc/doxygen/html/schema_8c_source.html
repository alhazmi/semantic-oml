153 static OMLSemDef *
154 sem_to_mpdef(const char *cs, int *ncs)
155 {
156  OMLSemDef *osdp=NULL,*last_osdp;
157  const char *csp;
158  int len = strlen(cs);
159  // Walk all characters of the string concepts
160  cs=(char*)find_charn(cs,'{',len);
161  while (cs&&*cs=='{')
162  {
163  last_osdp = (OMLSemDef*)oml_malloc(sizeof(OMLSemDef));
164  csp = (char*)find_charn(++cs, '|', --len);
165  last_osdp->subject = oml_strndup(cs,csp-cs);
166  len -= csp+1-cs;
167  cs = csp+1;
168  csp = (char*)find_charn(cs, '|', len);
169  last_osdp->predicate = oml_strndup(cs,csp-cs);
170  len -= csp+1-cs;
171  cs = csp+1;
172  csp = (char*)find_charn(cs, '}', len);
173  last_osdp->object = oml_strndup(cs,csp-cs);
174  len -= csp+1-cs;
175  cs = csp+1;
176  last_osdp->next = NULL;
177  add_oml_sem_def(&osdp,last_osdp);
178  (*ncs)++;
179  }
180  return osdp;
181 }

246 {
247  if (!meta) return NULL;
248 
249  struct schema *schema = NULL;
250  struct schema_field *f = NULL;
251  size_t fields_size = 0;
252  const char *p = meta;
253  char *q;
254 
255  schema = oml_malloc (sizeof (struct schema));
256  if (!schema) { goto exit; }
257  schema->nfields = 0;
258  schema->fields = NULL;
259 
260  /* Parse the schema index */
261  schema->index = strtol (p, &q, 0);
262  if (p == q) {
263  return NULL; /* no digits found */
264  }
265  p = q;
266 
267  /* Get the name of the schema */
268  p = (char*)skip_white (p);
269  q = (char*)find_white (p);
270  schema->name = oml_strndup (p, (q - p));
271  if (!schema->name) { goto exit; }
272  p = q;
273 
274  /* Parse the field definitions */
275  while (q && *q) {
276  p = (char*)skip_white (p);
277  q = (char*)find_white (p);
278  if (p != q) {
279  schema->nfields++;
280  fields_size += sizeof (struct schema_field);
281  /* We need this intermediary to be able to free the previous block on failure */
282  if (!(f = oml_realloc (schema->fields, fields_size))) {
283  goto exit;
284  }
285  schema->fields = f;
286  if (schema_field_from_meta (p, (q - p), &(schema->fields[schema->nfields-1])) == -1) {
287  goto exit;
288  }
289  p = q;
290  }
291  }
292  return schema;
293  exit:
294  if (schema) {
295  schema_free(schema);
296  }
297  return NULL;

390 /* Convert a schema into an SQL table creation statement.
391  *
392  * \param schema schema structure as created by, e.g., schema_from_meta()
393  * \param typemap function pointer to the database adapter's SQL to OML type converter function
394  * \return an oml_malloc'd MString containing the table-creation statement; the caller is responsible of freeing it
395  * \see oml_malloc, oml_free
396  */
397 MString*
398 schema_to_sql (const struct schema* schema, reverse_typemap o2t)
399 {
400  int n = 0;
401  int max = schema->nfields;
402 
403  if (max <= 0) {
404  logerror ("Tried to create SQL CREATE TABLE statement for schema with 0 columns\n");
405  return NULL;
406  }
407 
408  MString* mstr = mstring_create ();
409  if (mstr == NULL) {
410  logerror("Failed to create managed string for preparing SQL CREATE TABLE statement\n");
411  return NULL;
412  }
413 
414  /* Build SQL "CREATE TABLE" statement */
415  n += mstring_set (mstr, "CREATE TABLE \"");
416  n += mstring_cat (mstr, schema->name);
417  n += mstring_sprintf (mstr, "\" (oml_tuple_id %s, ", o2t (OML_DB_PRIMARY_KEY));
418  n += mstring_sprintf (mstr, "oml_sender_id %s, ", o2t (OML_INT32_VALUE));
419  n += mstring_sprintf (mstr, "oml_seq %s, ", o2t (OML_INT32_VALUE));
420  n += mstring_sprintf (mstr, "oml_ts_client %s, ", o2t (OML_DOUBLE_VALUE));
421  n += mstring_sprintf (mstr, "oml_ts_server %s", o2t (OML_DOUBLE_VALUE));
422 
423  int i = 0;
424  while (max > 0) {
425  OmlValueT type = schema->fields[i].type;
426  char *name = schema->fields[i].name;
427  const char* t = o2t (type);
428  if (!t) {
429  logerror("Unknown type in column '%s'\n", name);
430  goto fail_exit;
431  }
432  n += mstring_sprintf (mstr, ", \"%s\" %s", name, t);
433  i++; max--;
434  }
435  n += mstring_cat (mstr, ");");
436 
437  if (n != 0) {
438  /* mstring_* return -1 on error, with no error, the sum in n should be 0 */
439  goto fail_exit;
440  }
441  return mstr;
442 
443  fail_exit:
444  if (mstr) mstring_delete (mstr);
445  return NULL;
446 }

448 /*
449  * Check to see if "field" represents a metadata column. If so,
450  * check the type matches the expected type. If the type mismatches,
451  * return -1. If the type matches, return 0. If the field is not a
452  * metadata field, return 1.
453  */
454 static int
455 schema_check_metadata (struct schema_field *field)
456 {
457  struct schema_field metadata [] =
458  {
459  { "oml_tuple_id", OML_DB_PRIMARY_KEY },
460  { "oml_sender_id", OML_INT32_VALUE },
461  { "oml_seq", OML_INT32_VALUE },
462  { "oml_ts_client", OML_DOUBLE_VALUE },
463  { "oml_ts_server", OML_DOUBLE_VALUE },
464  };
465  size_t i;
466  for (i = 0; i < LENGTH (metadata); i++) {
467  if (!strcmp (metadata[i].name, field->name)) {
468  if (metadata[i].type != field->type) {
469  const char *expected = oml_type_to_s (metadata[i].type);
470  const char *received = oml_type_to_s (field->type);
471  logerror ("Existing table metadata type mismatch (%s expected %s, got %s)\n",
472  metadata[i].name, expected, received);
473  return -1;
474  } else return 0;
475  }
476  }
477  return 1;
478 }

489 struct schema*
490 schema_from_sql (const char *sql, OmlValueT (*t2o) (const char *s))
491 {
492  const char * const command = "CREATE TABLE ";
493  int command_len = strlen (command);
494  /* Check that it's a CREATE TABLE statement */
495  if (strncmp (sql, command, command_len)) return NULL;
496 
497  struct schema *schema = NULL;
498  struct schema_field *fields = NULL;
499  int nfields = 0;
500  size_t fields_size = 0;
501  char *name;
502  char *p = (char *)sql + command_len, *up;
503  char *q, *uq;
504 
505  p = (char*)skip_white (p);
506  q = (char*)find_white (p);
507  up = *p == '"' ? p+1 : p;
508  uq = *(q-1) == '"' ? q-1 : q;
509  name = oml_strndup (up, uq - up);
510  p = q;
511  p = (char*)skip_white (p);
512 
513  if (*p++ != '(') /* Opening paren of colspec */
514  goto exit;
515 
516  q = p;
517  while (q && *q && *q != ';') {
518  p = (char*)skip_white (p);
519  q = memchr (p, ',', strlen (p));
520  if (q == NULL) {
521  q = memchr (p, ')', strlen (p));
522  }
523  if (q != NULL) {
524  fields_size = (nfields + 1) * sizeof (struct schema_field);
525  struct schema_field *f = oml_realloc (fields, fields_size);
526  if (!f)
527  goto exit;
528  fields = f;
529  if (schema_field_from_sql (p, (q - p), &fields[nfields], t2o) == -1)
530  goto exit;
531  int n = schema_check_metadata (&fields[nfields]);
532  if (n == -1)
533  goto exit;
534  /* n == 0 means metadata column (must be skipped here) */
535  if (n == 0) {
536  oml_free (fields[nfields].name);
537  fields[nfields].name = NULL;
538  }
539  fields[nfields].concepts = NULL;
540  nfields += n;
541  p = ++q;
542  }
543  }
544 
545  schema = oml_malloc (sizeof (struct schema));
546  if (!schema) goto exit;
547 
548  schema->name = name;
549  schema->index = -1;
550  schema->fields = fields;
551  schema->nfields = nfields;
552  return schema;
553 
554  exit:
555  if (name) oml_free (name);
556  if (fields) {
557  int i = 0;
558  for (; i < nfields; i++)
559  if (fields[i].name)
560  oml_free (fields[i].name);
561  oml_free (fields);
562  }
563  oml_free (schema);
564  return NULL;
565 }

732 int
733 schema_diff (struct schema *s1, struct schema *s2)
734 {
735  int i=0, nfields = 0, diffn=0;
736  if (s1 == s2) return 0;
737  if (!s1 || !s2) return -1;
738  if (strcmp (s1->name, s2->name)) return -1;
739  if (s1->fields && s2->fields) {
740 
741  if (s1->nfields != s2->nfields) { diffn=1; }
742  nfields = (s1->nfields<s2->nfields)?s1->nfields:s2->nfields;
743 
744  for (i = 0; i < nfields; i++) {
745  if (strcmp (s1->fields[i].name, s2->fields[i].name) ||
746  s1->fields[i].type != s2->fields[i].type) {
747  return i+1;
748  }
749  }
750  if (diffn) {
751  /* The first differing field is the first one
752  * beyond the shortest schema's */
753  return nfields + 1;
754  }
755 
756  } else {
757  return -1;
758  }
759 
760  return 0;
761 }
