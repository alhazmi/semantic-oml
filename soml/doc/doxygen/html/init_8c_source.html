594 static void
595 install_close_handler(sighandler sig_hdl)
596 {
597  struct sigaction new_action, old_action;
598 
599  /* Set up the structure to specify the new action. */
600  new_action.sa_handler = sig_hdl;
601  sigemptyset (&new_action.sa_mask);
602  new_action.sa_flags = 0;
603 
604  sigaction (SIGINT, NULL, &old_action);
605  if (old_action.sa_handler != SIG_IGN) {
606  sigaction (SIGINT, &new_action, NULL);
607  }
608 
609  sigaction (SIGHUP, NULL, &old_action);
610  if (old_action.sa_handler != SIG_IGN) {
611  sigaction (SIGHUP, &new_action, NULL);
612  }
613 
614  sigaction (SIGTERM, NULL, &old_action);
615  if (old_action.sa_handler != SIG_IGN) {
616  sigaction (SIGTERM, &new_action, NULL);
617  }
618 
619  sigaction (SIGPIPE, NULL, &old_action);
620  if (old_action.sa_handler != SIG_IGN) {
621  sigaction (SIGPIPE, &new_action, NULL);
622  }

821  oml_free ((void*)port);

837  }

839  omlc_instance->first_writer = writer;
