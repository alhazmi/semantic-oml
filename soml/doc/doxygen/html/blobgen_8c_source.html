48 randgen (size_t *n)
49 {
50  long length;
51  int i;
52  uint8_t *data = NULL;
53  // 10000 bytes lee-way on short blobs
54  size_t max_size = longblob ? MAX_BLOB : USHRT_MAX - 10000;
55 
56  if (fixed_size > 0)
57  length = fixed_size;
58  else
59  while ((size_t)(length = random()) > max_size);
60 
61  data = malloc (length);
62 
63  if (data == NULL) {
64  perror("randgen()");
65  return NULL;
66  }
67

69  data[i] = (uint8_t)(random() & 0xFF);

82 write_blob_hex (int fd, void *blob, size_t n)
83 {
84  char hex [] = "0123456789ABCDEF";
85  unsigned int i = 0;
86  uint8_t *hexblob = malloc (2 * n);
87  int result = 0;
88  for (i = 0; i < n; i++) {
89  uint8_t *nybbles = (uint8_t*)hexblob + 2 * i;
90  uint8_t *data = (uint8_t*)blob + i;
91  nybbles[0] = hex[(*data >> 4) & 0xF];
92  nybbles[1] = hex[*data & 0xF];
93  }
94  result = write (fd, hexblob, 2 * n);
95  free (hexblob);
96  return result;

168 run (void)
169 {
170  int i = 0;
171  void *blob = NULL;
172  size_t blob_length = 0, totlength = 0;
173  char s[64];

178 
179  /* Piggyback on v[0] which should later contain a string */
180  omlc_set_string(v[0], "v1");
181  omlc_inject_metadata(mp, "k1", &v[0], OML_STRING_VALUE, NULL);
182  meta_to_file("k1", "v1", "blobmp", NULL);
183 
184  gettimeofday(&beg, NULL);
185  fprintf (stderr, "# blobgen: writing blobs:");
186  for (i = 0; samples != 0; i++, samples--) {
187  snprintf (s, sizeof(s)-1, "sample-%04d\n", i);
188  blob = randgen (&blob_length);
189  blob_to_file (i + 1, blob, blob_length);
190  totlength += blob_length;
191  fprintf (stderr, " %d (%dB)", i, (int)blob_length);
192  omlc_set_string (v[0], s);
193  omlc_set_int32 (v[1], i);
194  omlc_set_blob (v[2], blob, blob_length);
195  omlc_inject (mp, v);
196  fflush(stderr);
197  usleep(interval);
198  }
199  gettimeofday(&end, NULL);
200  deltaT = difftv(end, beg);
201  fprintf (stderr, " (%d injects and %dB in %fs: %fips, %fBps).\n", i, (int)totlength, deltaT,
202  (double)i/deltaT, (double)totlength/deltaT);
203 
204  omlc_set_string(v[0], "v2");
205  omlc_inject_metadata(mp, "k2", &v[0], OML_STRING_VALUE, "blob");
206  meta_to_file("k2", "v2", "blobmp", "blob");
207  omlc_inject_metadata(mp, "k1", &v[0], OML_STRING_VALUE, NULL);
208  meta_to_file("k1", "v2", "blobmp", NULL);
