105 int
106 client_realloc_tables (ClientHandler *self, int ntables)
107 {
108  if (!self || ntables <= 0) {
109  return -1;
110  }
111 
112  if (ntables > self->table_count) {
113  int error = 0;
114  DbTable **new_tables = oml_realloc (self->tables, ntables*sizeof(DbTable*));
115  int *new_so = oml_realloc (self->seqno_offsets, ntables*sizeof(int));
116  OmlValue **new_vv = oml_realloc (self->values_vectors, ntables*sizeof(OmlValue*));
117  int *new_vv_counts = oml_realloc (self->values_vector_counts, ntables * sizeof (int));
118 
119  if (!new_tables || !new_so || !new_vv || !new_vv_counts) {
120  logdebug ("%s: Failed to allocate memory for %d more client tables (current %d)\n",
121  self->name, (ntables - self->table_count), self->table_count);
122  // Don't free anything because whatever got successfully oml_realloc'd is still ok
123  error = -1;
124  }
125 
126  /* Keep whatever succeeded */
127  if (new_tables) {
128  self->tables = new_tables;
129  /* Initialise all new tables to NULL so we know they are not allocated yet */
130  memset(&self->tables[self->table_count], 0, (ntables - self->table_count) * sizeof(DbTable*));
131  }
132  if (new_so) self->seqno_offsets = new_so;
133  if (new_vv) self->values_vectors = new_vv;
134  if (new_vv_counts) self->values_vector_counts = new_vv_counts;
135 
136  /* If the values vectors succeeded, we need to create the new ones */
137  if (new_vv && new_vv_counts) {
138  int i = 0;
139  for (i = self->table_count; i < ntables; i++) {
140  self->values_vectors[i] = oml_calloc (DEF_NUM_VALUES, sizeof (OmlValue));
141  if (self->values_vectors[i]) {
142  self->values_vector_counts[i] = DEF_NUM_VALUES;
143  oml_value_array_init(self->values_vectors[i], self->values_vector_counts[i]);
144  } else {
145  self->values_vector_counts[i] = -1;
146  logdebug ("%s: Could not allocate memory for values vector for table index %d\n", self->name, i);
147  error = -1;
148  }
149  }
150  }
151  if (!error)
152  self->table_count = ntables;
153  return error;
154  }
155 
156  return 0;
157 }

300 {

302  if (!schema) {
303  logerror ("%s: Failure parsing schema '%s'; disconnecting client.\n", self->name, value);
304  self->state = C_PROTOCOL_ERROR;
305  return;
306  }
307 
308  char *invalid = NULL;
309  if (!validate_schema_names (schema, &invalid)) {
310  logerror ("%s: Invalid name '%s' in schema '%s'\n", self->name, invalid, value);
311  self->state = C_PROTOCOL_ERROR;
312  schema_free (schema);
313  return;
314  }
315 
316  int idx = schema->index;
317  loginfo("%s: New MS schema %s\n", self->name, value); /* Value contains the index */
318 
319  DbTable* table = database_find_or_create_table(self->database, schema);
320  if (table == NULL) {
321  logerror("%s: Can't find table '%s' or client schema '%s' doesn't match any of the existing tables.\n",
322  self->name, schema->name, value);
323  self->state = C_PROTOCOL_ERROR;
324  schema_free (schema);
325  return;
326  }
327  schema_free (schema);
328 
329  if (client_realloc_tables (self, idx + 1) == -1) {
330  logerror ("%s: Failed to allocate memory for table index %d '%s'\n",
331  self->name, idx, schema->name);
332  return;
333  }
334 
335  if (self->tables[idx] != NULL) {
336  logwarn("%s: Replacing existing MS schema %s with new schema %s\n",
337  self->name, self->tables[idx]->schema->name, value);
338  }
339  self->tables[idx] = table;
340 
341  /* Reallocate the values vector if this schema has more columns than can fit already. */
342  if (client_realloc_values (self, idx, table->schema->nfields) == -1) {
343  logwarn ("%s: Could not allocate values vector of size %d for table index %d '%s'\n",
344  self->name, table->schema->nfields, idx, schema->name);
345  }

580 static int
581 process_header(ClientHandler* self, MBuffer* mbuf)
582 {
583  char* line;
584  int len;
585 
586  if (read_line(&line, &len, mbuf) == 0) {
587  return 0;
588  }
589 
590  if (len == 0) {
591  // empty line denotes separator between header and body
592  int skip_count = mbuf_find_not (mbuf, '\n');
593  mbuf_read_skip (mbuf, skip_count + 1);
594  mbuf_consume_message (mbuf);
595  self->state = self->content;
596  client_event_report(self, "Ready", "");
597  loginfo("%s: Client %s ready to send data\n", self->name, self->event->name);
598  return 0;
599  }
600 
601  // separate key from value (check for ':')
602  char* value = line;
603  int count = 0;
604  while (*(value) != ':' && count < len) {
605  value++;
606  count++;
607  }
608 
609  if (*value == ':') {
610  *value++ = '\0';
611  while (*(value) == ' ' && count < len) {
612  value++;
613  count++;
614  }
615  mbuf_read_skip (mbuf, len + 1);
616  process_meta(self, line, value);
617  } else {
618  logerror("%s: Malformed meta line in header: '%s'\n", self->name, line);
619  self->state = C_PROTOCOL_ERROR;
620  }
621 
622  // process_meta() might have signalled protocol error, so we have to check here.
623  if (self->state == C_PROTOCOL_ERROR)
624  return 0;
625  else
626  return 1; // still in header
627 }

640 static void
641 process_bin_data_message(ClientHandler* self, OmlBinaryHeader* header)
642 {
643  double ts;
644  int table_index;
645  int seqno;
646  struct schema *schema;
647  int i, ki = -1, vi = -1, si = -1;
648  DbTable *table;
649  MBuffer* mbuf = self->mbuf;
650  OmlValue *v;
651  int count;
652 
653  ts = header->timestamp;
654  table_index = header->stream;
655  seqno = header->seqno;
656 
657  if (header->stream < 0 || table_index >= self->table_count) {
658  logwarn("%s(bin): Table index %d out of bounds, discarding sample %d\n",
659  self->name, table_index, seqno);
660  return;
661  }
662 
663  ts += self->time_offset;
664  table = self->tables[table_index];
665  if (NULL == table) {
666  if (0 == table_index) {
667  /* If this is schema 0, there is a chance the ClientHandler
668  * doesn't know about it yet; find it */
669  table = database_find_table(self->database,
670  "_experiment_metadata");
671  client_realloc_tables(self, 1); /* Make sure we have space for 1 */
672  client_realloc_values(self, 0, table->schema->nfields);
673  self->tables[table_index] = table;
674  } else {
675  logerror("%s(bin): Undefined table index %d\n", self->name, table_index);
676  return;
677  }
678  }
679 
680  v = self->values_vectors[table_index];
681  /* These OmlValue are properly initialised by client_realloc_values,
682  * however, the schema might have been redefined sinc last time */
683  count = self->values_vector_counts[table_index];
684  oml_value_array_reset(v, count);
685  count = unmarshal_measurements(mbuf, header, v, count);
686 
687  schema = table->schema;
688  if (count<-100) {
689  logerror("%s(bin): An error occured during unmarshalling (%d)\n",
690  self->name, count);
691  return;
692  } else if (schema->nfields != count) {
693  logerror("%s(bin): Data item number mismatch for schema '%s' (expected %d, got %d)\n",
694  self->name, schema->name, schema->nfields, count - 3);
695  return;
696  }
697  mbuf_consume_message (mbuf);
698 
699  if (0 == table_index) { /* Stream 0: Metadata */
700  logdebug("%s(bin): Client sending metadata at %f\n", self->name, ts);
701 
702  /* For future-proofness: find fields with actual names "key" and "value",
703  * regardless of how many there are .*/
704  for (i = 0; i < schema->nfields; i++) {
705  logdebug3("%s(bin): Found field %s at index %d in schema %s\n",
706  self->name, schema->fields[i].name, i, schema->name);
707  if (!strcmp(schema->fields[i].name, "key")) {
708  ki = i;
709  } else if (!strcmp(schema->fields[i].name, "value")) {
710  vi = i;
711  } else if (!strcmp(schema->fields[i].name, "subject")) {
712  si = i;
713  }
714  }
715  if (ki<0 || vi<0 || si<0) {
716  logerror("%s(bin): Trying to process metadata from a schema without 'subject', 'key' or 'value' fields\n", self->name);
717  return;
718 
719  } else if (oml_value_get_type(&v[si]) != OML_STRING_VALUE ) {
720  logwarn("%s(bin): Expecting metadata, but subject is not a string (%d), ignoring\n",
721  self->name, oml_value_get_type(&v[si]));
722 
723  } else if (strcmp(".", omlc_get_string_ptr(*oml_value_get_value(&v[si])))) {
724  logdebug("%s(bin): Metadata subject '%s' is not the root, ignoring\n",
725  self->name, omlc_get_string_ptr(*oml_value_get_value(&v[si])));
726 
727  } else if (oml_value_get_type(&v[ki]) != OML_STRING_VALUE ) {
728  logwarn("%s(bin): Expecting metadata, but key is not a string (%d), ignoring\n",
729  self->name, oml_value_get_type(&v[ki]));
730 
731  } else if (oml_value_get_type(&v[vi]) != OML_STRING_VALUE ) {
732  logwarn("%s(bin): Expecting metadata, but value for key %s is not a string (%d), ignoring\n",
733  self->name, omlc_get_string_ptr(*oml_value_get_value(&v[ki])),
734  oml_value_get_type(&v[vi]));
735 
736  } else if(process_meta(self,
737  omlc_get_string_ptr(*oml_value_get_value(&v[ki])),
738  omlc_get_string_ptr(*oml_value_get_value(&v[vi]))) <= 0) {
739  logdebug("%s(bin): No need to store metadata separately\n", self->name);
740  return;
741  }
742  }
743 
744  logdebug("%s(bin): Inserting data into table index %d '%s' (seqno=%d, ts=%f)\n",
745  self->name, table_index, table->schema->name, seqno, ts);
746  self->database->insert(self->database, table, self->sender_id, header->seqno,
747  ts, self->values_vectors[table_index], count);
748 }

756 static int
757 process_bin_message(ClientHandler* self, MBuffer* mbuf)
758 {
759  char *out;
760  int res;
761  OmlBinaryHeader header;
762 
763  res = bin_find_sync(mbuf);
764  if(res>0) {
765  logwarn("%s(bin): Skipped %d bytes of data searching for a new message\n", self->name, res);
766  } else if (res == -1 && mbuf_rd_remaining(mbuf)>=2) {
767  logdebug("%s(bin): Invalid or no message found in binary packet\n", self->name);
768  if(o_log_level_active(O_LOG_DEBUG4)) {
769  out = to_octets(mbuf_rdptr(mbuf), mbuf_rd_remaining(mbuf));
770  logdebug("%s(bin): Packet follows\n%s", self->name, out);
771  oml_free(out);
772  }
773  return 0;
774  }
775 
776  res = unmarshal_init(mbuf, &header);
777  if (res == 0) {
778  logwarn("%s(bin): Could not find message header\n", self->name);
779  return 0;
780  } else if (res < 0 && mbuf_fill(mbuf)>0) {
781  // not enough data
782  logdebug2("%s(bin): Not enough data (%dB) for a new measurement yet (at least %dB missing)\n",
783  self->name, mbuf_rd_remaining(mbuf), -res);
784  return 0;
785  }
786  switch (header.type) {
787  case OMB_DATA_P:
788  case OMB_LDATA_P:
789  process_bin_data_message(self, &header);
790  if (self->state != C_BINARY_DATA)
791  return 0;
792  break;
793  default:
794  logwarn("%s(bin): Ignoring unsupported message type '%d'\n", self->name, header.type);
795  /* XXX: Assume we could read the full header, just skip it
796  * FIXME: We might have to skip the data too
797  self->state = C_PROTOCOL_ERROR;
798  */
799  return 0;
800  }
801 
802  return 1;
803 }

817 static void
818 process_text_data_message(ClientHandler* self, char** msg, int count)
819 {
820  double ts;
821  int table_index;
822  int seqno;
823  struct schema *schema;
824  int i, ki = -1, vi = -1, si = -1;
825  DbTable *table;
826  OmlValue *v;
827 
828  if (count < 3) {
829  return;
830  }
831 
832  ts = atof(msg[0]);
833  table_index = atol(msg[1]);
834  seqno = atol(msg[2]);
835 
836  if (table_index < 0 || table_index >= self->table_count) {
837  logwarn("%s(txt): Table index %d out of bounds, discarding sample %d\n",
838  self->name, table_index, seqno);
839  return;
840  }
841 
842  ts += self->time_offset;
843  table = self->tables[table_index];
844  if (NULL == table) {
845  if (0 == table_index) {
846  /* If this is schema 0, there is a chance the ClientHandler
847  * doesn't know about it yet; find it */
848  table = database_find_table(self->database,
849  "_experiment_metadata");
850  client_realloc_tables(self, 1); /* Make sure we have space for 1 */
851  client_realloc_values(self, 0, table->schema->nfields);
852  self->tables[table_index] = table;
853  } else {
854  logerror("%s(txt): Undefined table index %d\n", self->name, table_index);
855  return;
856  }
857  }
858 
859  schema = table->schema;
860  if (count<-100) {
861  logerror("%s(txt): An error occured during unmarshalling (%d)\n",
862  self->name, count);
863  return;
864  } else if (schema->nfields != count - 3) { /* Ignore first 3 elements */
865  logerror("%s(txt): Data item number mismatch for schema '%s' (expected %d, got %d)\n",
866  self->name, schema->name, schema->nfields, count - 3);
867  return;
868  }
869 
870  if (0 == table_index) { /* Stream 0: Metadata */
871  logdebug("%s(txt): Client sending metadata at %f\n", self->name, ts);
872 
873  /* For future-proofness: find fields with actual names "key" and "value",
874  * regardless of how many there are .*/
875  for (i = 0; i < schema->nfields; i++) {
876  logdebug3("%s(txt): Found field %s at index %d in schema %s\n",
877  self->name, schema->fields[i].name, i, schema->name);
878  if (!strcmp(schema->fields[i].name, "key")) {
879  ki = i + 3; /* Ignore first 3 elements */
880  } else if (!strcmp(schema->fields[i].name, "value")) {
881  vi = i + 3; /* Ignore first 3 elements */
882  } else if (!strcmp(schema->fields[i].name, "subject")) {
883  si = i + 3; /* Ignore first 3 elements */
884  }
885  }
886  if (ki<0 || vi<0 || si<0) {
887  logerror("%s(txt): Trying to process metadata from a schema without 'subject', 'key' or 'value' fields\n", self->name);
888  return;
889 
890  } else if (strcmp(".", msg[si])) {
891  logdebug("%s(txt): Metadata subject '%s' is not the root, not processing\n",
892  self->name, msg[si]);
893 
894  } else if(process_meta(self, msg[ki], msg[vi]) <=0 ) {
895  logdebug2("%s(txt): No need to store metadata separately\n", self->name);
896  return;
897  }
898  }
899 
900  if (self->values_vector_counts[table_index] < schema->nfields) {
901  logerror("%s(txt): Not enough OmlValues (%d) to hold received data (%d)\n",
902  self->name, self->values_vector_counts[table_index], schema->nfields);
903  return;
904  }
905 
906  v = self->values_vectors[table_index];
907  /* These OmlValue are properly initialised by client_realloc_values,
908  * however, the schema might have been redefined sinc last time */
909  oml_value_array_reset(v, count);
910 
911  for (i=0; i < schema->nfields; i++) {
912  oml_value_set_type(&v[i], schema->fields[i].type);
913  if (oml_value_from_s (&v[i], msg[i+3]) == -1) {
914  logerror("%s(txt): Error converting value of type %d from string '%s'\n", self->name, oml_value_get_type(v), msg[i+3]);
915  return;
916  }
917  }
918 
919  logdebug("%s(txt): Inserting data into table index %d '%s' (seqno=%d, ts=%f)\n",
920  self->name, table_index, table->schema->name, seqno, ts);
921  self->database->insert(self->database, table, self->sender_id, seqno,
922  ts, self->values_vectors[table_index], count - 3); /* Ignore first 3 elements */
923 }

931 static int
932 process_text_message(ClientHandler* self, MBuffer* mbuf)
933 {
934  char* line;
935  int len;
936 
937  while (C_TEXT_DATA == self->state) {
938  if (read_line(&line, &len, mbuf) == 0) {
939  return 0;
940  }
941  mbuf_read_skip(mbuf, len+1);
942  mbuf_consume_message(mbuf);
943 
944  // split line into array
945  char* a[DEF_NUM_VALUES];
946  int a_size = 0;
947  char* p = line;
948  int rem = len;
949  int lastempty = 0;
950 
951  while (rem > 0 || lastempty) {
952  char* param = p;
953  lastempty = 0;
954  for (; rem > 0; rem--, p++) {
955  if (*p == '\t') {
956  *(p++) = '\0';
957  rem--;
958  // Set if there was an empty string at the end of the line
959  lastempty = (0 == rem && '\0' == *(p-1) && !lastempty);
960  break;
961  }
962  }
963  a[a_size++] = param;
964  if (a_size >= DEF_NUM_VALUES) {
965  logerror("%s(txt): Too many parameters (%d>=%d) in sample '%s'\n", self->name, a_size, DEF_NUM_VALUES, line);
966  return 0;
967  }
968  }
969  /* XXX: This message belongs in process_text_data_message(),
970  * however putting it here allows to access line, for nicer logging*/
971  if (a_size < 3) {
972  logerror("%s(txt): Not enough parameters (%d<3) in sample '%s'\n", self->name, a_size, line);
973  return 0;
974  }
975  process_text_data_message(self, a, a_size);
976  }
977  /* Never reached */
978  return 0;
979 }

987  void
988 client_callback(SockEvtSource* source, void* handle, void* buf, int buf_size)
989 {
990  char *in;
991  ClientHandler* self = (ClientHandler*)handle;
992  MBuffer* mbuf = self->mbuf;
993 
994  logdebug2("%s(%s): Received %d bytes of data\n",
995  source->name,
996  client_state_to_s (self->state),
997  buf_size);
998 
999  if(o_log_level_active(O_LOG_DEBUG4)) {
1000  in = to_octets(buf, buf_size);
1001  logdebug2("%s(%s): Received new packet\n%s\n",
1002  source->name, client_state_to_s (self->state), in);
1003  oml_free(in);
1004  }
1005 
1006  int result = mbuf_write (mbuf, buf, buf_size);
1007 
1008  if (result == -1) {
1009  logerror("%s: Failed to write message from client into message buffer\n",
1010  source->name);
1011  return;
1012  }
1013 
1014 process:
1015  switch (self->state)
1016  {
1017  case C_HEADER:
1018  while (process_header(self, mbuf));
1019  if (self->state != C_HEADER) {
1020  //finished header, let someone else process rest of buffer
1021  goto process;
1022  }
1023  break;
1024 
1025  case C_BINARY_DATA:
1026  while (process_bin_message(self, mbuf));
1027  break;
1028 
1029  case C_TEXT_DATA:
1030  while (process_text_message(self, mbuf));
1031  break;
1032 
1033  case C_PROTOCOL_ERROR:
1034  // Protocol error: close the client connection
1035  logerror("%s: Fatal error, disconnecting client\n",
1036  source->name);
1037  client_event_report(self, "Disconnect", "C_PROTOCOL_ERROR");
1038  client_handler_free (self);
1039  /*
1040  * Protocol error --> no need to repack buffer, so just return;
1041  */
1042  return;
1043  default:
1044  logerror("%s: Unknown client state %d\n", source->name, self->state);
1045  mbuf_clear (mbuf);
1046  return;
1047  }
1048 
1049  if (self->state == C_PROTOCOL_ERROR)
1050  goto process;
1051 
1052  // move remaining buffer content to beginning
1053  mbuf_repack_message (mbuf);
1054  logdebug2("%s: Buffer repacked to %d bytes\n", source->name, mbuf_fill(mbuf));
1055 }
1062 void
1063 status_callback(SockEvtSource* source, SocketStatus status, int errcode, void* handle)
1064 {
1065  ClientHandler* self = (ClientHandler*)handle;
1066 
1067  logdebug("%s: Socket status changed to %s (%d); error code is %d\n",
1068  source->name, socket_status_string (status), status, errcode);
1069 
1070  const char *event = "Disconnect";
1071  const char *message = socket_status_string(status);
1072  switch (status)
1073  {
1074  case SOCKET_WRITEABLE:
1075  break;
1076  case SOCKET_CONN_CLOSED:
1077  /* Client closed the connection */
1078  client_event_report(self, event, message);
1079  loginfo("%s: Client %s closed connection\n", self->name, source->name);
1080  client_handler_free (self);
1081  break;
1082  case SOCKET_IDLE:
1083  /* Server dropped idle connection */
1084  client_event_report(self, event, message);
1085  loginfo("%s: Client %s dropped due to idleness\n", self->name, source->name);
1086  client_handler_free (self);
1087  break;
1088  case SOCKET_UNKNOWN:
1089  case SOCKET_CONN_REFUSED:
1090  case SOCKET_DROPPED:
1091  default:
1092  client_event_report(self, event, message);
1093  logwarn ("%s: Client '%s' received unhandled condition %s (%d)\n", source->name,
1094  self->name, socket_status_string (status), status);
1095  break;
1096  }
1097 }
