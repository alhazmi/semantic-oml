153 static int
154 open_socket(OmlNetOutStream* self)
155 {
156  struct sigaction new_action, old_action;
157 
158  if(self->socket) {
159  socket_free(self->socket);
160  self->socket = NULL;
161  }
162  if (strcmp(self->protocol, "tcp") == 0) {
163  Socket* sock;
164  if ((sock = socket_tcp_out_new(self->dest, self->host, self->service)) == NULL) {
165  return 0;
166  }
167 
168  self->socket = sock;
169  self->header_written = 0;
170  } else {
171  logerror("%s: Unsupported transport protocol '%s'\n", self->dest, self->protocol);
172  return 0;
173  }
174 
175  // Catching SIGPIPE signals if the associated socket is closed
176  // TODO: Not exactly sure if this is completely right for all application situations.
177  new_action.sa_handler =signal_handler;
178  sigemptyset (&new_action.sa_mask);
179  new_action.sa_flags = 0;
180 
181  sigaction (SIGPIPE, NULL, &old_action);
182  /* XXX: Shouldn't we set up the handler ONLY if the old one is SIG_IGN? */
183  if (old_action.sa_handler != SIG_IGN) {
184  sigaction (SIGPIPE, &new_action, NULL);
185  }
186 
187  return 1;
188 }

197 static size_t
198 net_stream_write(OmlOutStream* hdl, uint8_t* buffer, size_t length, uint8_t* header, size_t header_length)
199 {
200  OmlNetOutStream* self = (OmlNetOutStream*)hdl;
201 
202  /* Initialise the socket the first time */
203  while (self->socket == NULL) {
204  logdebug ("%s: Connecting to server\n", self->dest);
205  if (!open_socket(self)) {
206  logdebug("%s: Connection attempt failed\n", self->dest);
207  return 0;
208  }
209  }
210 
211  /* If the underlying socket as registered a disconnection, it will reconnect on its own
212  * however, we need to check it to make sure we send the headers before anything else */
213  if(socket_is_disconnected(self->socket)) {
214  self->header_written = 0;
215  }
216 
217  size_t count;
218  if (! self->header_written) {
219  if(o_log_level_active(O_LOG_DEBUG4)) {
220  char *out = to_octets(header, header_length);
221  logdebug("%s: Sending header %s\n", self->dest, out);
222  oml_free(out);
223  }
224  if ((count = socket_write(self, header, header_length)) < header_length) {
225  // TODO: This is not completely right as we end up rewriting the same header
226  // if we can only partially write it. At this stage we think this is too hard
227  // to deal with and we assume it doesn't happen.
228  if (count > 0) {
229  // PANIC
230  logwarn("%s: Only wrote parts of the header; this might cause problem later on\n", self->dest);
231  }
232  return 0;
233  }
234  self->header_written = 1;
235  }
236  if(o_log_level_active(O_LOG_DEBUG4)) {
237  char *out = to_octets(buffer, length);
238  logdebug("%s: Sending data %s\n", self->dest, out);
239  oml_free(out);
240  }
241  count = socket_write(self, buffer, length);
242  return count;
243 }
