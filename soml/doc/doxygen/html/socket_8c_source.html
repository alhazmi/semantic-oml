198 Socket*
199 socket_in_new(const char* name, const char* node, const char* service, int is_tcp)
200 {
201  SocketInt *self = NULL, *list = NULL;
202  char nameserv[SOCKNAMELEN], *namestr = NULL;
203  size_t namestr_size;
204  struct addrinfo hints, *results, *rp;
205  int ret;
206 
207  *nameserv = 0;
208  memset(&hints, 0, sizeof(struct addrinfo));
209 
210  hints.ai_socktype = SOCK_STREAM;
211  hints.ai_protocol = IPPROTO_TCP; /* FIXME: We should check self->is_tcp */
212  hints.ai_flags= AI_PASSIVE;
213  int val = 1;
214 
215  if((ret=getaddrinfo(node, service, &hints, &results))) {
216  o_log(O_LOG_ERROR, "socket(%s): Error resolving %s:%s: %s\n",
217  name, node, service, gai_strerror(ret));
218  return NULL;
219  }
220 
221  for (rp = results; rp != NULL; rp = rp->ai_next) {
222  sockaddr_get_name((sockaddr_t*)rp->ai_addr, rp->ai_addrlen, nameserv, SOCKNAMELEN);
223  namestr_size = strlen(name) + strlen(nameserv + 2);
224  namestr = oml_realloc(namestr, namestr_size);
225  snprintf(namestr, namestr_size, "%s-%s", name, nameserv);
226  o_log(O_LOG_DEBUG, "socket(%s): Binding to %s (AF %d, proto %d)\n",
227  name, nameserv, rp->ai_family, rp->ai_protocol);
228 
229  if (NULL == (self = (SocketInt*)socket_new(namestr, is_tcp))) {
230  o_log(O_LOG_WARN, "socket(%s): Could allocate Socket to listen on %s: %s\n",
231  self->name, nameserv, strerror(errno));
232 
233  } else if(0 > (self->sockfd =
234  socket(rp->ai_family, rp->ai_socktype, rp->ai_protocol))) {
235  /* It's OK if some AFs fail, we check that in the end */
236  o_log(O_LOG_DEBUG, "socket(%s): Could not create socket to listen on %s: %s\n",
237  self->name, nameserv, strerror(errno));
238  socket_free((Socket*)self);
239 
240  } else if(0 != setsockopt(self->sockfd, SOL_SOCKET, SO_REUSEADDR, &val, sizeof(int))){
241  o_log(O_LOG_WARN, "socket(%s): Could not set option SO_REUSEADDR on socket to listen on %s: %s\n",
242  self->name, nameserv, strerror(errno));
243  close(self->sockfd);
244  /* XXX: Not optimal: we could reuse the Socket */
245  socket_free((Socket*)self);
246 
247  } else if (0 != bind(self->sockfd, rp->ai_addr, rp->ai_addrlen)) {
248  o_log(O_LOG_WARN, "socket(%s): Error binding socket to listen on %s: %s\n",
249  self->name, nameserv, strerror(errno));
250  close(self->sockfd);
251  /* XXX: Not optimal: we could reuse the Socket */
252  socket_free((Socket*)self);
253 
254  } else {
255  /* The last connected AFs (hence less important according to GAI) are
256  * now at the front, but it's not a big deal as we treat them all
257  * equally (i.e, we're going to listen on them all */
258  self->next = (Socket*)list;
259  list = self;
260  self = NULL;
261  }
262 
263  }
264 
265  if (namestr) { oml_free(namestr); }
266  freeaddrinfo(results);
267 
268  if (NULL == list) {
269  o_log(O_LOG_ERROR, "socket(%s): Could not create any socket to listen on port %s\n",
270  name, service);
271  }
272 
273  return (Socket*)list;
274 }

283 static int
284 s_connect(SocketInt* self)
285 {
286  char name[SOCKNAMELEN];
287  struct addrinfo hints;
288  int ret;
289 
290  *name = 0;
291 
292  memset(&hints, 0, sizeof(struct addrinfo));
293  /* XXX: This should be pulled up when we support UDP and/or multicast */
294  hints.ai_socktype = SOCK_STREAM;
295  hints.ai_protocol = IPPROTO_TCP; /* FIXME: We should check self->is_tcp */
296 
297  if (!self->dest || !self->service) {
298  o_log(O_LOG_ERROR, "socket(%s): destination or service missing. Is this an outgoing socket?\n",
299  self->name);
300  return 0;
301  }
302 
303  if (!self->rp) {
304  o_log(O_LOG_DEBUG, "socket(%s): Resolving %s:%s\n",
305  self->name, self->dest, self->service);
306  if ((ret=getaddrinfo(self->dest, self->service, &hints, &self->results))) {
307  o_log(O_LOG_ERROR, "socket(%s): Error resolving %s:%s: %s\n",
308  self->name, self->dest, self->service, gai_strerror(ret));
309  return 0;
310  }
311  self->rp = self->results;
312  }
313 
314  for (; self->rp != NULL; self->rp = self->rp->ai_next) {
315  sockaddr_get_name((sockaddr_t*)self->rp->ai_addr, self->rp->ai_addrlen,
316  name, SOCKNAMELEN);
317  o_log(O_LOG_DEBUG, "socket(%s): Connecting to %s (AF %d, proto %d)\n",
318  self->name, name, self->rp->ai_family, self->rp->ai_protocol);
319 
320  /* If we are here, any old socket is invalid, clean them up and try again */
321  if(self->sockfd >= 0) {
322  o_log(O_LOG_DEBUG2, "socket(%s): FD %d already open, closing...\n",
323  self->name, self->sockfd);
324  close(self->sockfd);
325  }
326 
327  if(0 > (self->sockfd =
328  socket(self->rp->ai_family, self->rp->ai_socktype, self->rp->ai_protocol))) {
329  o_log(O_LOG_DEBUG, "socket(%s): Could not create socket to %s %s\n",
330  self->name, name, strerror(errno));
331 
332  } else {
333  if (nonblocking_mode) {
334  fcntl(self->sockfd, F_SETFL, O_NONBLOCK);
335  }
336 
337  if (0 != connect(self->sockfd, self->rp->ai_addr, self->rp->ai_addrlen)) {
338  o_log(O_LOG_DEBUG, "socket(%s): Could not connect to %s: %s\n",
339  self->name, name, strerror(errno));
340 
341  } else {
342  if (!nonblocking_mode) {
343  o_log(O_LOG_DEBUG, "socket(%s): Connected to %s\n",
344  self->name, name);
345  }
346  self->is_disconnected = 0;
347 
348  /* Get ready to try the next addrinfo in case this one failed */
349  //self->rp = self->rp->ai_next;
350  return 1;
351 
352  }
353  }
354  }
355  o_log(O_LOG_WARN, "socket(%s): Failed to connect\n",
356  self->name, self->dest, self->service);
357 
358  freeaddrinfo(self->results);
359  self->results = NULL;
360  return 0;
361 }

527 int
528 socket_sendto(Socket* socket, char* buf, int buf_size)
529 {
530  SocketInt *self = (SocketInt*)socket;
531  int sent;
532 
533 
534  if (self->is_disconnected) {
535  if(!s_connect(self)) {
536  return 0;
537  }
538 
539  } else if((sent = recv(self->sockfd, NULL, 0, MSG_DONTWAIT)) == 0) {
540  /* Test that the server kept the other side of the connection alive */
541  if (!(sent < 0 && EAGAIN == errno)) {
542  /* EAGAIN is the expected case: connection alive but no data,
543  * everything else is a problem */
544  logwarn("socket(%s): Server appears to have closed the connection\n", self->name);
545 
546  self->is_disconnected = 1;
547  /* We want higher levels to be aware of this disconnection */
548  return 0;
549  }
550 
551  }
552 
553  if ((sent = sendto(self->sockfd, buf, buf_size, MSG_NOSIGNAL,
554  &(self->servAddr.sa),
555  sizeof(self->servAddr.sa_stor))) < 0) {
556  if (errno == EPIPE || errno == ECONNRESET) {
557  // The other end closed the connection.
558  self->is_disconnected = 1;
559  o_log(O_LOG_ERROR, "socket(%s): The remote peer closed the connection: %s\n",
560  self->name, strerror(errno));
561  return 0;
562  } else if (errno == ECONNREFUSED) {
563  self->is_disconnected = 1;
564  o_log(O_LOG_DEBUG, "socket(%s): Connection refused, trying next AI\n",
565  self->name);
566  self->rp = self->rp->ai_next;
567  return 0;
568  } else if (errno == EINTR) {
569  o_log(O_LOG_WARN, "socket(%s): Sending data interrupted: %s\n",
570  self->name, strerror(errno));
571  return 0;
572  } else {
573  o_log(O_LOG_ERROR, "socket(%s): Sending data failed: %s\n",
574  self->name, strerror(errno));
575  }
576  return -1;
577  }
578  return sent;
579 }
