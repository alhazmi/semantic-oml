148 int
149 omlc_inject(OmlMP *mp, OmlValueU *values)
150 {
151  OmlMStream* ms;
152  OmlValue v;
153  int i;
154 
155  if (NULL == omlc_instance || omlc_instance->start_time <= 0) {
156  logerror("Cannot inject samples prior to calling omlc_init and omlc_start\n");
157  return -1;
158  }
159  if (mp == NULL || values == NULL) {
160  return -1;
161  }
162 
163  LOGDEBUG("Injecting data into MP '%s'\n", mp->name);
164 
165  oml_value_init(&v);
166  if (mp_lock(mp) == -1) {
167  logwarn("Cannot lock MP '%s' for injection\n", mp->name);
168  return -1;
169  }
170 
171  uint64_t written = 0;
172  uint64_t dropped = 0;
173  for (ms = mp->streams; ms; ms = ms->next) {
174  LOGDEBUG("Filtering MP '%s' data into MS '%s'\n", mp->name, ms->table_name);
175  OmlFilter* f = ms->filters;
176  for (; f != NULL; f = f->next) {
177 
178  /* FIXME: Should validate this indexing */
179  oml_value_set(&v, &values[f->index], mp->param_defs[f->index].param_types);
180 
181  f->input(f, &v);
182  }
183  omlc_ms_process(ms);
184  written += ms->written;
185  dropped += ms->dropped;
186  for (i=0; i<ms->nwriters; i++) {
187  dropped += bw_nlost_reset(ms->writers[i]->bufferedWriter);
188  }
189  }
190  mp_unlock(mp);
191  oml_value_reset(&v);
192 
193  /* do we need to send client instrumentation? */
194  if(mp != omlc_instance->client_instr && omlc_instance->instr_interval) {
195  time_t now;
196  time(&now);
197  if(omlc_instance->instr_time + omlc_instance->instr_interval <= now) {
198  omlc_instance->instr_time = now; /* Make sure we don't loop */
199  omlc_inject_client_instr(written, dropped, xmemnew(), xmemfreed(), xmembytes(), xmaxbytes());
200  }
201  }
202 
203  return 0;
204 }
