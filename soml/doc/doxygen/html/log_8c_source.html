137 static void
138 _o_log(time_t now, int level, char *msg)
139 {
140  const int label_max = LENGTH(log_labels)-1;
141  int label_index;
142  struct tm* ltime;
143  char now_str[20], dbglvl[] = { 0, 0 };
144 
145  label_index = level - O_LOG_ERROR; /* O_LOG_ERROR is negative */
146  label_index = (label_index < 0) ? 0 : label_index;
147  label_index = (label_index > label_max) ? label_max : label_index;
148 
149  if (level > O_LOG_DEBUG) {
150  *dbglvl = '0' + (level - O_LOG_INFO);
151  }
152 
153  if (o_log_simplified == o_log_function && stderr == logfile) {
154  /* This logic should clearly go into o_log_simplified,
155  * but as it is variadic, we have little option to manipulate its parameters */
156  o_log_function(level, "%s%s\t%s", log_labels[label_index], dbglvl, msg);
157 
158  } else {
159  ltime = localtime(&now);
160  strftime(now_str, 20, "%b %d %H:%M:%S", ltime);
161  o_log_function(level, "%s %s%s\t%s", now_str, log_labels[label_index], dbglvl ,msg);
162  }
163 }

182 static void
183 o_vlog(int log_level, const char* fmt, va_list va)
184 {
185  static char b1[LOG_BUF_LEN], b2[LOG_BUF_LEN], *new_log=NULL, *last_log=NULL, *tmp;
186  static int last_level = O_LOG_INFO;
187  static time_t last_time = (time_t)0;
188  static uint64_t nseen = 0;
189  static uint64_t exponent = INIT_LOG_EXPONENT;
190  time_t now;
191 
192  if (!o_log_level_active(log_level)) { return; }
193 
194  time(&now);
195 
196  if (!new_log || !last_log || last_time == (time_t)-1) {
197  /* Initialisation of static arrays */
198  *b1=0;
199  *b2=0;
200  new_log = b1;
201  last_log = b2;
202  }
203 
204  vsnprintf(new_log, LOG_BUF_LEN, fmt, va);
205 
206  if (difftime(now, last_time) < MAX_MESSAGE_RATE &&
207  !strncmp(new_log, last_log, LOG_BUF_LEN) &&
208  /* Same message and level at too high a rate */
209  last_level == log_level) {
210 
211  if(++nseen > exponent) {
212  snprintf(new_log, LOG_BUF_LEN,
213  "Message repeated %" PRIu64 " time%s: %s", nseen, nseen>1?"s":"", last_log);
214  _o_log(now, log_level, new_log);
215 
216  last_time = now;
217  nseen = 0;
218 
219  if (exponent < (INT64_MAX/2)) { exponent *= 2; }
220  }
221 
222  } else {
223  /* Give a final count of all the previously unreported
224  * similar messages before printing the new one */
225  if (nseen > 1) {
226  snprintf(last_log, LOG_BUF_LEN,
227  "Previous message repeated %" PRIu64 " times\n", nseen);
228  _o_log(last_time, last_level, last_log);
229 
230  } else if (nseen > 0) {
231  /* If it was only repeated once, simply show it */
232  _o_log(last_time, last_level, last_log);
233 
234  }
235 
236  /* New message */
237  _o_log(now, log_level, new_log);
238 
239  last_time = now;
240  last_level = log_level;
241  nseen = 0;
242  exponent = INIT_LOG_EXPONENT;
243 
244  tmp = new_log;
245  new_log = last_log;
246  last_log = tmp;
247 
248  }
249 }
