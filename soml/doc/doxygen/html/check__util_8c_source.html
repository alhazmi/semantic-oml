36 OmlValueU*
37 make_vector (void* v, OmlValueT type, int n)
38 {
39  OmlValueU* result = (OmlValueU*)malloc (n * sizeof (OmlValueU));
40 
41  if (result)
42  {
43  int i = 0;
44  omlc_zero_array(result, n);
45 
46  switch (type)
47  {
48  case OML_LONG_VALUE:
49  for (i = 0; i < n; i++)
50  omlc_set_long(result[i], ((long*)v)[i]);
51  break;
52  case OML_DOUBLE_VALUE:
53  for (i = 0; i < n; i++)
54  omlc_set_double(result[i], ((double*)v)[i]);
55  break;
56  case OML_INT32_VALUE:
57  for (i = 0; i < n; i++)
58  omlc_set_int32(result[i], ((int32_t*)v)[i]);
59  break;
60  case OML_UINT32_VALUE:
61  for (i = 0; i < n; i++)
62  omlc_set_uint32(result[i], ((uint32_t*)v)[i]);
63  break;
64  case OML_INT64_VALUE:
65  for (i = 0; i < n; i++)
66  omlc_set_int64(result[i], ((int64_t*)v)[i]);
67  break;
68  case OML_UINT64_VALUE:
69  for (i = 0; i < n; i++)
70  omlc_set_uint64(result[i], ((uint64_t*)v)[i]);
71  break;
72  default:
73  // Can't do non-numeric types yet
74  return NULL;
75  }
76 
77  return result;
78  }
79  else
80  return NULL;
81 }

119 int
120 vector_values_check (OmlValue* values, OmlValueU* expected, OmlValueT type, int n)
121 {
122  double epsilon = 1e-9;
123  int i = 0;
124  for (i = 0; i < n; i++) {
125  switch (type) {
126  case OML_LONG_VALUE:
127  if (omlc_get_long(*oml_value_get_value(&values[i])) != omlc_get_long(expected[i]))
128  return 0;
129  break;
130  case OML_DOUBLE_VALUE:
131  if (fabs(omlc_get_double(*oml_value_get_value(&values[i])) - omlc_get_double(expected[i])) > epsilon)
132  return 0;
133  break;
134  case OML_STRING_VALUE:
135  if (omlc_get_string_length(*oml_value_get_value(&values[i])) != omlc_get_string_length(expected[i]))
136  return 0;
137  if (strcmp (omlc_get_string_ptr(*oml_value_get_value(&values[i])), omlc_get_string_ptr(expected[i])) != 0)
138  return 0;
139  break;
140  case OML_INT32_VALUE:
141  if (omlc_get_int32(*oml_value_get_value(&values[i])) != omlc_get_int32(expected[i]))
142  return 0;
143  break;
144  case OML_UINT32_VALUE:
145  if (omlc_get_uint32(*oml_value_get_value(&values[i])) != omlc_get_uint32(expected[i]))
146  return 0;
147  break;
148  case OML_INT64_VALUE:
149  if (omlc_get_int64(*oml_value_get_value(&values[i])) != omlc_get_int64(expected[i]))
150  return 0;
151  break;
152  case OML_UINT64_VALUE:
153  if (omlc_get_uint64(*oml_value_get_value(&values[i])) != omlc_get_uint64(expected[i]))
154  return 0;
155  break;
156  default:
157  return 0; // Fail on unknown value types
158  }
159  }
160  return 1;
161 }

163 char *vector_values_stringify(OmlValue *v, int n)
164 {
165  static char str[256];
166  int i = 0, used = 0;
167 
168  do {
169  if (i>0) {

171  used = strlen(str);

174  used = strlen(str);
175  } while (++i<n && used < sizeof(str));

180 char *vector_values_ut_stringify(OmlValueU* values, OmlValueT type, int n)
181 {
182  OmlValue v;

184  int i = 0, used = 0;
185 
186  oml_value_init(&v);
187 
188  do {
189  if (i>0) {
190  strncat(str, " ", sizeof(str) - used);
191  used = strlen(str);
192  }
193  oml_value_set(&v, &values[i], type);
194  oml_value_to_s(&v, &str[used], sizeof(str) - used);
195  used = strlen(str);
196  } while (++i<n && used < sizeof(str));
197 
198 
199  oml_value_reset(&v);

234 void
235 run_filter_test (TestData* test_data, OmlFilter* f)
236 {
237  int i = 0, j = 0;
238  int count = 0;
239  OmlTestWriter w;
240  OmlValue v;
241 
242  oml_value_init(&v);
243 
244  w.out = test_writer_out;
245 
246  for (i = 0; i < test_data->count; i++) {
247  TestVector* input = test_data->inputs[i];
248  TestVector* output = test_data->outputs[i];
249  oml_value_set_type(&v, input->type);
250  for (j = 0; j < input->length; j++, count++)
251  {
252  int res;
253  /* XXX: is this really legal? */
254  v.value = input->vector[j];
255  res = f->input(f, &v);
256  fail_unless (res == 0);
257  }
258 
259  f->output (f, (OmlWriter*)&w);
260  f->newwindow (f);
261 
262  fail_unless (w.count == output->length);
263  fail_unless (vector_type_check (w.values, output->type, output->length));
264  fail_unless (vector_values_check (w.values, output->vector, output->type, output->length),
265  "Output mismatch in test vector %d [%s] for filter '%s': expected [%s] but got [%s]",
266  i,
267  vector_values_ut_stringify(input->vector, input->type, input->length),
268  f->name,
269  vector_values_ut_stringify(output->vector, output->type, output->length),
270  vector_values_stringify(w.values, w.count));
271  }
272  /* Ask for data when nothing more has happened*/
273  f->output (f, (OmlWriter*)&w);
274  fail_unless (w.count == test_data->outputs[0]->length,
275  "Filter with empty window didn't return the right amount of data items (%d instead of %d)",
276  w.count, test_data->outputs[0]->length);
277 
278  oml_value_reset(&v);
279 }
