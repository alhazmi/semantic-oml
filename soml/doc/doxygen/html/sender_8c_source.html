123  return -1;

125 
126  struct header *header = client->headers;
127  while (header) {

129  if (client_send_header (client, header) == -1)

165  continue; // Haven't finished receiving the headers
166 
167 
168  pthread_mutex_unlock (&client->mutex);

173  } else {
174  logdebug ("Failed to connect to downstream: %s\n", strerror (errno));
175  sleep (1); // Retry every second.
176  continue;
177  }
178  }
179  pthread_mutex_lock (&client->mutex);
180  if (client->sender_connected && client_send_headers (client) == -1) {
181  logdebug ("Something failed while sending headers: %s\n", strerror (errno));
182  client->sender_connected = 0;
183  continue;
184  }
185  }
186 
187  while (client->messages->length > 0 && client->sender_connected) {
188  struct msg_queue_node *head = msg_queue_head (client->messages);
189  struct cbuffer_cursor cursor = head->cursor;
190  size_t length = head->msg->length;
191  size_t count = 0;
192  int result = 0;
193 
194  while (length > 0) {
195  char *buf = cbuf_cursor_pointer (&cursor);
196  size_t page_remaining = cbuf_cursor_page_remaining (&cursor);
197  size_t to_write = length < page_remaining ? length : page_remaining;
198 
199  logdebug ("Seqno: %d Sending %d bytes (fill=%d, remain=%d)\n", head->msg->seqno, to_write, cursor.page->fill, page_remaining);
200  pthread_mutex_unlock (&client->mutex);
201  result = send (client->send_socket, buf, to_write, 0);
202  pthread_mutex_lock (&client->mutex);
203 
204  if (result == -1) {
205  logdebug ("send(2) returned -1 (%s)\n", strerror (errno));
206  if (sigpipe_flag == 1) {
207  logdebug ("sigpipe_flag is 1\n");
208  sigpipe_flag = 0;
209  }
210  client->sender_connected = 0;
211  break;
212  } else if (result < (int)to_write) {
213  logdebug ("Wrote less than expected \n");
214  }
215  cbuf_advance_cursor (&cursor, result);
216  length -= result;
217  count += result;
218  }
219  if (result != -1) {
220  cbuf_consume_cursor (&head->cursor, head->msg->length);
221  msg_queue_remove (client->messages);
222  }
223  }
224

226  loginfo ("Client disconnected and all pending measurements have been sent; shutting down this client\n");
227  shutdown (client->send_socket, SHUT_RDWR);
228  sleep (1);

230  session_remove_client (session, client);
231  client_free (client);
232  pthread_exit (NULL);
233  }
234  } else if (session->state == ProxyState_PAUSED && client->sender_connected) {
235  shutdown (client->send_socket, SHUT_RDWR);
236  sleep (1);
237  close (client->send_socket);
238  client->sender_connected = 0;
239  }
240  }
241 }
