115 int
116 dba_table_create_meta (Database *db, const char *name)
117 {
118  int i = 0;
119  for (i = 0; i < LENGTH (meta_tables); i++) {
120  if (strcmp (meta_tables[i].name, name) == 0) {
121  if (meta_tables[i].sql) {
122  logdebug("%s:%s: Creating default table %s from SQL '%s'\n",
123  db->backend_name, db->name,
124  name, meta_tables[i].sql);
125  return db->stmt (db, meta_tables[i].sql);
126  } else if (meta_tables[i].schema) {
127  logdebug("%s:%s: Creating default table %s from schema '%s'\n",
128  db->backend_name, db->name,
129  name, meta_tables[i].schema);
130  return dba_table_create_from_meta(db, meta_tables[i].schema);
131  } else {
132  logwarn ("%s:%s: Default definition found for %s, but it is not complete\n", name);
133  }
134  }
135  }
136  return -1;
137 }
