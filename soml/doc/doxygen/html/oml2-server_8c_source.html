178 static void sighandler(int signum)
179 {
180  switch(signum) {
181  case SIGINT:
182  case SIGTERM:
183  loginfo("Received signal %d, cleaning up and exiting\n", signum);
184  eventloop_terminate(signum);
185  break;
186  case SIGUSR1:
187  eventloop_report(O_LOG_INFO);
188  break;
189  default:
190  logwarn("Received unhandled signal %d\n", signum);
191  }
192 }

195 typedef void (*sh_t) (int);
200 static void signal_install(sh_t handler)
201 {
202  struct sigaction sa;
203 
204  sa.sa_handler = (void(*) (int))handler;
205  sigemptyset (&sa.sa_mask);
206  sa.sa_flags = 0;
207 
208  if(sigaction(SIGTERM, &sa, NULL))
209  logwarn("Unable to install SIGTERM handler: %s\n", strerror(errno));
210  if(sigaction(SIGINT, &sa, NULL))
211  logwarn("Unable to install SIGINT handler: %s\n", strerror(errno));
212  if(sigaction(SIGUSR1, &sa, NULL))
213  logwarn("Unable to install SIGUSR1 handler: %s\n", strerror(errno));
214 }

236 static void drop_privileges (const char *uidstr, const char *gidstr)
237 {
238  if (gidstr && !uidstr)
239  die ("--gid supplied without --uid\n");
240 
241  if (uidstr) {
242  struct passwd *passwd = getpwnam (uidstr);
243  gid_t gid;
244 
245  if (!passwd)
246  die ("User '%s' not found\n", uidstr);
247  if (!gidstr)
248  gid = passwd->pw_gid;
249  else {
250  struct group *group = getgrnam (gidstr);
251  if (!group)
252  die ("Group '%s' not found\n", gidstr);
253  gid = group->gr_gid;
254  }
255 
256  struct group *group = getgrgid (gid);
257  const char *groupname = group ? group->gr_name : "??";
258  gid_t grouplist[] = { gid };
259 
260  if (setgroups (1, grouplist) == -1)
261  die ("Couldn't restrict group list to just group '%s': %s\n", groupname, strerror (errno));
262 
263  if (setgid (gid) == -1)
264  die ("Could not set group id to '%s': %s", groupname, strerror (errno));
265 
266  if (setuid (passwd->pw_uid) == -1)
267  die ("Could not set user id to '%s': %s", passwd->pw_name, strerror (errno));
268 
269  if (setuid (0) == 0)
270  die ("Tried to drop privileges but we seem able to become superuser still!\n");
271  }
272 }

292 int main(int argc, const char **argv)
293 {
294  int c;
295 
296 #ifdef HAVE_LIBPQ
297  char *pass_replace = "--pg-pass=WITHHELD", *conninfo_replace = "--pg-connect=WITHHELD";
298 #endif
299  char *vir_pass_replace = "--vir-pass=WITHHELD";
300  char semantic = 0;
301 
302  oml_setup(&argc, argv);
303 
304  poptContext optCon = poptGetContext(NULL, argc, (const char**) argv, options, 0);
305 
306  while ((c = poptGetNextOpt(optCon)) >= 0) {
307  switch (c) {
308  case 'v':
309  printf(V_STRING, VERSION);
310  printf("OML Protocol V%d--%d\n", MIN_PROTOCOL_VERSION, MAX_PROTOCOL_VERSION);
311  printf(COPYRIGHT);
312  return 0;
313  }
314  }
315 
316 #ifdef HAVE_LIBPQ
317  /* Cleanup command line to avoid showing credentials in ps(1) output, amongst
318  * others.
319  *
320  * XXX: This is a poor man's security measure, as this creates a race
321  * condition where, prior to doing the following, the credentials are still
322  * visible to everybody.
323  */
324  if (pg_pass || pg_conninfo) {
325  for(c=1; c<argc; c++) {
326  if(pg_pass && !strncmp(argv[c],"--pg-pass", 9)) {
327  argv[c] = pass_replace;
328  }
329 
330  if(pg_conninfo && !strncmp(argv[c],"--pg-connect", 12)) {
331  argv[c] = conninfo_replace;
332  }
333  }
334  }
335  for(c=1; c<argc; c++) {
336  if (vir_user && !strncmp(argv[c],"--vir-user", 10))
337  semantic = 1;
338  else if (vir_pass && !strncmp(argv[c],"--vir-pass", 10))
339  {
340  argv[c] = conninfo_replace;
341  semantic = 1;
342  }
343  else if (vir_host && !strncmp(argv[c],"--vir-host", 10))
344  semantic = 1;
345  else if (vir_port && !strncmp(argv[c],"--vir-port", 10))
346  semantic = 1;
347  else if (fus_host && !strncmp(argv[c],"--fus-host", 10))
348  semantic = 1;
349  else if (fus_port && !strncmp(argv[c],"--fus-port", 10))
350  semantic = 1;
351  else if (fus_namespace && !strncmp(argv[c],"--fus-namespace", 15))
352  semantic = 1;
353  }
354 #endif /* HAVE_LIBPQ */
355  logging_setup (logfile_name, log_level);
356 
357  if (c < -1) {
358  die ("%s: %s\n", poptBadOption (optCon, POPT_BADOPTION_NOALIAS), poptStrerror (c));
359  }
360 
361  if (semantic)
362  {
363  loginfo(V_STRING_UAM, VERSION);
364  loginfo("Semantic-OML Protocol V%d--%d\n", MIN_PROTOCOL_VERSION, MAX_PROTOCOL_VERSION);
365  loginfo(COPYRIGHT_UAM);
366  }
367  else
368  {
369  loginfo(V_STRING, VERSION);
370  loginfo("OML Protocol V%d--%d\n", MIN_PROTOCOL_VERSION, MAX_PROTOCOL_VERSION);
371  loginfo(COPYRIGHT);
372  }
373 
374  eventloop_init();
375  eventloop_set_socket_timeout(socket_timeout);
376 
377  Socket* server_sock;
378  server_sock = socket_server_new("server", NULL, listen_service, on_connect, NULL);
379 
380  if (!server_sock) {
381  die ("Failed to create listening socket for service %s\n", listen_service);
382  }
383 
384  drop_privileges (uidstr, gidstr);
385 
386  /* Important that this comes after drop_privileges(). */
387  if(database_setup_backend(dbbackend)) {
388  die("Failed to setup database backend '%s'\n", dbbackend);
389  }
390 
391  signal_setup();
392 
393  hook_setup();
394 
395  eventloop_run();
396 
397  signal_cleanup();
398 
399  hook_cleanup();
400 
401  oml_cleanup();
402 
403  oml_memreport(O_LOG_INFO);
404 
405  return 0;
406 }
