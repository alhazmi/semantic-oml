103 size_t
104 file_stream_write(OmlOutStream* hdl, uint8_t* buffer, size_t length, uint8_t* header, size_t header_length)
105 {
106  OmlFileOutStream* self = (OmlFileOutStream*)hdl;
107 
108  /* The header can be NULL, but header_length MUST be 0 in that case */
109  assert(header || !header_length);
110 
111  if (!self) return -1;
112  FILE* f = self->f;
113  if (f == NULL) return -1;
114 
115  if (! self->header_written) {
116  size_t count;
117  if ((count = _file_stream_write(f, header, header_length)) < header_length) {
118  // TODO: This is not completely right as we end up rewriting the same header
119  // if we can only partially write it. At this stage we think this is too hard
120  // to deal with and we assume it doesn't happen.
121  if (count > 0) {
122  // PANIC
123  logerror("Only wrote part of the header. May screw up further processing\n");
124  }
125  return 0;
126  }
127  self->header_written = 1;
128  }
129  size_t count = _file_stream_write(f, buffer, length);
