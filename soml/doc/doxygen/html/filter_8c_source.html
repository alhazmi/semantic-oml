64 static void*
65 thread_start(void* handle)
66 {
67  OmlMStream* ms = (OmlMStream*)handle;
68  OmlMP* mp = ms->mp;
69  useconds_t usec = (useconds_t)(1000000 * ms->sample_interval);
70  int status = 0;
71 
72  while (1) {
73  usleep(usec);
74  if (!mp_lock(mp)) {
75  if (!mp->active) {
76  mp_unlock(mp);
77  return NULL; // we are done
78  }
79 
80  status = filter_process(ms);
81  mp_unlock(mp);
82  }
83 
84  if (status == -1) {
85  return NULL; // Fatal error --> exit thread
86  }
87 
88  }
89 }

102 int
103 filter_process(OmlMStream* ms)
104 {
105  struct timeval tv;
106  double now;
107  int i;
108  OmlFilter *f;
109  OmlWriter *writer;
110 
111  /* Get the time as soon as possible */
112  gettimeofday(&tv, NULL);
113 
114  if (ms == NULL || omlc_instance == NULL || ms->writers == NULL) {
115  logerror("Could not process filters because of null measurement stream, instance or writers array\n");
116  return -1;
117  }
118 
119  now = tv.tv_sec - omlc_instance->start_time + 0.000001 * tv.tv_usec;
120  ms->seq_no++;
121 
122  for (i=0; i<ms->nwriters; i++) {
123  writer = ms->writers[i];
124 
125  if (writer == NULL) {
126  logwarn("%s: Sending data NULL writer (at %d)\n", ms->table_name, i);
127 
128  } else {
129  /* Be aware that row_start is obtaining a lock on the writer
130  * which is released in row_end. Always ensure that row_end is
131  * called, even if there is a problem somewhere along the way.
132  * \see oml_writer_row_start, oml_writer_out, oml_writer_row_end
133  */
134  if(writer->row_start(writer, ms, now) == 1)
135  ms->written++;
136  else
137  ms->dropped++;
138 
139  f = ms->firstFilter;
140  for (; f != NULL; f = f->next) {
141  f->output(f, writer);
142  }
143  writer->row_end(writer, ms);
144  }
145  }
146 
147  f = ms->firstFilter;
148  for (; f != NULL; f = f->next) {
149  f->newwindow(f);
150  }
151  ms->sample_size = 0;
152 
153  return 0;
154 }
