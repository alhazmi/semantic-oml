257 int eventloop_run()
258 {
259  int i;
260  self.stopping = 0;
261  self.force_stop = 0;
262  self.start = self.now = self.last_reaped = time(NULL);
263  while (!self.stopping || (self.size>0 && !self.force_stop)) {
264  // Check for active timers
265  int timeout = -1;
266  TimerInt* t = self.timers;
267  while (t != NULL) {
268  if (t->is_active) {
269  int delta = 1000 * (t->due_time - self.now);
270  if (delta < 0) delta = 0; // overdue
271  if (delta < timeout || timeout < 0) timeout = delta;
272  }
273  t = t->next;
274  }
275  if (timeout != -1)
276  o_log(O_LOG_DEBUG3, "EventLoop: Timeout = %d\n", timeout);
277 
278  if (self.fds_dirty)
279  if (update_fds()<1 && timeout < 0) /* No FD nor timeout */
280  continue;
281  o_log(O_LOG_DEBUG4, "EventLoop: About to poll() on %d FDs with a timeout of %ds\n", self.size, timeout);
282  /* for(i=0; i < self.size; i++) {
283  o_log(O_LOG_DEBUG4, "EventLoop: FD %d->%s\n", self.fds[i].fd, self.fds_channels[i]->name);
284  } */
285 
286  int count = poll(self.fds, self.size, timeout);
287  self.now = time(NULL);
288 
289  if (count < 1) {
290  o_log(O_LOG_DEBUG4, "EventLoop: Timeout\n");
291  } else {
292  // Check sockets
293  i = 0;
294  o_log(O_LOG_DEBUG4, "EventLoop: Got events\n");
295  for (; i < self.size; i++) {
296  Channel* ch = self.fds_channels[i];
297  if (self.fds[i].revents & POLLERR) {
298  char buf[32];
299  SocketStatus status;
300  int len;
301 
302  if ((len = recv(self.fds[i].fd, buf, 32, 0)) <= 0) {
303  switch (errno) {
304  case ECONNREFUSED:
305  status = SOCKET_CONN_REFUSED;
306  break;
307  default:
308  status = SOCKET_UNKNOWN;
309  if (!ch->status_cbk) {
310  o_log(O_LOG_ERROR, "EventLoop: While reading from socket '%s': (%d) %s\n",
311  ch->name, errno, strerror(errno));
312  }
313  }
314  eventloop_socket_activate((SockEvtSource*)ch, 0);
315  do_status_callback (ch, status, errno);
316  } else {
317  o_log(O_LOG_ERROR, "EventLoop: Expected error on socket '%s' but read '%s'\n", ch->name, buf);
318  }
319  } else if (self.fds[i].revents & POLLHUP) {
320  eventloop_socket_activate((SockEvtSource*)ch, 0);
321 
322  /* Client closed the connection, but there might still be bytes
323  for us to read from our end of the connection. */
324  int len;
325  int fd = self.fds[i].fd;
326  char buf[MAX_READ_BUFFER_SIZE];
327  do {
328  if (fd == 0) {
329  len = read(fd, buf, MAX_READ_BUFFER_SIZE);
330  } else {
331  len = recv(fd, buf, 512, 0);
332  }
333  if (len > 0) {
334  o_log(O_LOG_DEBUG3, "EventLoop: Received last %i bytes\n", len);
335  do_read_callback (ch, buf, len);
336  }
337  } while (len > 0);
338  do_status_callback (ch, SOCKET_CONN_CLOSED, 0);
339  } else if (self.fds[i].revents & POLLIN) {
340  char buf[MAX_READ_BUFFER_SIZE];
341  if (ch->read_cbk) {
342  int len;
343  int fd = self.fds[i].fd;
344  if (fd == 0) {
345  // stdin
346  len = read(fd, buf, MAX_READ_BUFFER_SIZE);
347  } else {
348  // socket
349  len = recv(fd, buf, 512, 0);
350  }
351  ch->last_activity = self.now;
352  if (len > 0) {
353  o_log(O_LOG_DEBUG3, "EventLoop: Received %i bytes\n", len);
354  do_read_callback (ch, buf, len);
355  } else if (len == 0 && ch->socket != NULL) { // skip stdin
356  // closed down
357  eventloop_socket_activate((SockEvtSource*)ch, 0);
358  do_status_callback (ch, SOCKET_CONN_CLOSED, 0);
359  } else if (len < 0) {
360  if (errno == ENOTSOCK) {
361  o_log(O_LOG_ERROR,
362  "EventLoop: Monitored socket '%s' is now invalid; "
363  "removing from monitored set\n",
364  ch->name);
365  eventloop_socket_remove ((SockEvtSource*)ch);
366  } else {
367  o_log(O_LOG_ERROR, "EventLoop: Unrecognized read error not handled (errno=%d)\n",
368  errno);
369  }
370  }
371  } else {
372  do_monitor_callback (ch);
373  }
374  } else if (ch->is_shutting_down) {
375  /* The socket was shutting down, and nothing new has appeared;
376  * We flushed the buffers, mark it as removable */
377  eventloop_socket_release((SockEvtSource*)ch);
378  }
379 
380  if (self.fds[i].revents & POLLOUT) {
381  do_status_callback(ch, SOCKET_WRITEABLE, 0);
382  if (0 != ch->last_activity) {
383  /* If we track the activity of this socket */
384  ch->last_activity = self.now;
385  }
386  }
387 
388  if (self.fds[i].revents & POLLNVAL) {
389  o_log(O_LOG_WARN, "EventLoop: socket '%s' invalid, deactivating...\n", ch->name);
390  eventloop_socket_activate((SockEvtSource*)ch, 0);
391  do_status_callback(ch, SOCKET_DROPPED, 0);
392  }
393 
394  /* We reap idle channels as we go through the list. XXX: There might
395  * be a corner case where all FDs are already used, and some of them
396  * idle, however a new a new connection (on one listening socket early
397  * in the list) would be dropped before cleanup triggered by its
398  * arrival freed the resources it needs (from the idle sockets further
399  * towards the end of the list. See #959.*/
400  if (ch->last_activity != 0 &&
401  self.socket_timeout > 0 &&
402  self.now - ch->last_activity > self.socket_timeout) {
403  o_log(O_LOG_DEBUG2, "EventLoop: Socket '%s' idle for %ds, reaping...\n", ch->name, self.now - ch->last_activity);
404  do_status_callback(ch, SOCKET_IDLE, 0);
405  }
406  }
407  for (i = 0; i < self.size; i++) {
408  Channel* ch = self.fds_channels[i];
409  if (ch->is_removable)
410  eventloop_socket_remove ((SockEvtSource*)ch);
411  }
412  }
413  if (timeout >= 0) {
414  // check timers
415  TimerInt* t = self.timers;
416  while (t != NULL) {
417  if (t->is_active) {
418  if (t->due_time <= self.now) {
419  // fires
420  o_log(O_LOG_DEBUG2, "EventLoop: Timer '%s' fired\n", t->name);
421  if (t->callback) t->callback((TimerEvtSource*)t, t->handle);
422 
423  if (t->is_periodic) {
424  while ((t->due_time += t->period) < self.now) {
425  // should really only happen during debugging
426  o_log(O_LOG_WARN, "EventLoop: Skipped timer period for '%s'\n",
427  t->name);
428  }
429  } else {
430  t->is_active = 0;
431  }
432  }
433  }
434  t = t->next;
435  }
436  }
437  }
438  return self.stopping;
439 }

859 static int update_fds(void)
860 {
861  Channel* ch = self.channels;
862  int i = 0;
863 
864  while (ch != NULL) {
865  if (ch->is_active) {
866  if (self.length <= i) {
867  // Need to increase size of fds array
868  int l = (self.length > 0 ? 2 * self.length : DEF_FDS_LENGTH);
869  self.fds = (struct pollfd *)realloc(self.fds, l * sizeof(struct pollfd));
870  self.fds_channels = (Channel **)realloc(self.fds_channels, l * sizeof(Channel*));
871  self.length = l;
872  }
873  self.fds[i].fd = ch->fds_fd;
874  self.fds[i].events = ch->fds_events;
875  self.fds_channels[i] = ch;
876  i++;
877  }
878  ch = ch->next;
879  }
880  o_log(O_LOG_DEBUG, "EventLoop: %d active channel%s\n", i, i>1?"s":"");
881 
882  self.size = i;
883  self.fds_dirty = 0;
884 
885  return i;
886 }

898 static void terminate_fds(void)
899 {
900  Channel *ch = self.channels, *next;
901 
902  while (ch != NULL) {
903  next = ch->next;
904  o_log(O_LOG_DEBUG4, "EventLoop: Terminating channel %s\n", ch->name);
905  if (!ch->is_active ||
906  socket_is_disconnected(ch->socket) ||
907  socket_is_listening(ch->socket)) {
908  o_log(O_LOG_DEBUG3, "EventLoop: Releasing listening channel %s\n", ch->name);
909  eventloop_socket_release((SockEvtSource*)ch);
910  } else if (self.force_stop) {
911  o_log(O_LOG_DEBUG3, "EventLoop: Closing down %s\n", ch->name);
912  eventloop_socket_release((SockEvtSource*)ch);
913  socket_close(ch->socket);
914  } else {
915  o_log(O_LOG_DEBUG3, "EventLoop: Shutting down %s\n", ch->name);
916  socket_shutdown(ch->socket);
917  ch->is_shutting_down = 1;
918  }
919  ch = next;
920  }
921 
922  update_fds();
923 }

968 static void do_status_callback (Channel *ch, SocketStatus status, int error)
969 {
970  if (ch->status_cbk && !ch->is_removable) {
971  ch->status_cbk ((SockEvtSource*)ch, status, error, ch->handle);
972 
973  } else if (!ch->status_cbk) {

975

977  case SOCKET_WRITEABLE:
978  break;
979  case SOCKET_CONN_CLOSED:
980  case SOCKET_CONN_REFUSED:
981  case SOCKET_DROPPED:
982  case SOCKET_IDLE:
983  o_log(O_LOG_DEBUG, "EventLoop: Closing socket '%s' due to status %d\n", ch->name, status);
984  eventloop_socket_release((SockEvtSource*)ch);
985  break;
986  case SOCKET_UNKNOWN:
987  default:
988  o_log(O_LOG_WARN, "EventLoop: Unexpected status on socket '%s': %d\n", ch->name, status);
989  break;
990  }
