57 int
58 read_header (Client *self, MBuffer *mbuf)
59 {
60  char *line;
61  int len;
62  struct header *header;
63 
64  if (self == NULL || mbuf == NULL) {
65  logerror ("read_header(): NULL client or input buffer.\n");
66  return 0;
67  }
68 
69  if (read_line (&line, &len, mbuf) == -1)
70  return 0;
71 
72  if (len == 0) {
73  // Empty line denotes separator between header and body
74  int skip_count = mbuf_find_not (mbuf, '\n');
75  if (skip_count == -1) {
76  while (mbuf_find (mbuf, '\n') != (size_t)-1)
77  mbuf_read_skip (mbuf, 1);
78  } else {
79  mbuf_read_skip (mbuf, skip_count);
80  }
81  self->state = C_CONFIGURE;
82  return 0;
83  }
84 
85  header = header_from_string (line, len);
86  mbuf_read_skip (mbuf, len + 1);
87  if (header == NULL) {
88  /* Could be protocol error (not ':'), memory alloc error, or unknown tag */
89  self->state = C_PROTOCOL_ERROR;
90  return -1;
91  }
92 
93  /* Store the header for later processing */
94  header->next = self->headers;
95  self->headers = header;
96  if (header->tag != H_NONE) {
97  self->header_table[header->tag] = header;
98  }
99 
100  return 1; // still in header
101 }

144 void
145 proxy_message_loop (const char *client_id, Client *client, void *buf, size_t size)
146 {
147  MBuffer *mbuf = client->mbuf;
148  struct header *header;
149  struct oml_message msg;
150  int result;
151  size_t message_length;
152 
153  result = mbuf_write (mbuf, buf, size);
154  if (result == -1) {
155  logerror ("'%s': Failed to write message from client into message buffer. Data is being lost!\n",
156  client_id);
157  return;
158  }
159 
160  loop:
161  switch (client->state) {
162  case C_HEADER:
163  /* Read headers until out of header data to read*/
164  while (read_header (client, mbuf) == 1);
165  if (client->state != C_HEADER)
166  goto loop;
167  break;
168  case C_CONFIGURE:
169  if (client->header_table[H_DOMAIN] == NULL ||
170  client->header_table[H_CONTENT] == NULL ||
171  client->header_table[H_DOMAIN]->tag == H_NONE ||
172  client->header_table[H_CONTENT]->tag == H_NONE) {
173  if (client->headers == NULL) {
174  logdebug ("Headers NULL!");
175  }
176  client->state = C_PROTOCOL_ERROR;
177  } else {
178  client->domain = client->header_table[H_DOMAIN]->value;
179  client->content = content_from_string (client->header_table[H_CONTENT]->value);
180  if (client->content == CONTENT_NONE)
181  client->state = C_PROTOCOL_ERROR;
182  }
183  if (client->state != C_PROTOCOL_ERROR) {
184  logdebug ("%s\n", client->domain);
185  logdebug ("%d\n", client->content);
186  } else {
187  logdebug ("Can't write out domain and content because of protocol error in input\n");
188  logdebug ("Input is: '%s'\n", buf);
189  }
190  for (header = client->headers; header != NULL; header = header->next) {
191  logdebug ("HEADER: '%s' : '%s'\n",
192  tag_to_string(header->tag),
193  header->value);
194  }
195  switch (client->content){
196  case CONTENT_TEXT:
197  client->msg_start = text_read_msg_start;
198  break;
199  case CONTENT_BINARY:
200  client->msg_start = bin_read_msg_start;
201  break;
202  default:
203  /* Default is set when client is created (client_new()) */
204  logerror ("Client content is not TEXT or BINARY\n");
205  break;
206  }
207  mbuf_consume_message (mbuf); // Next message starts after the headers.
208  client->state = C_DATA;
209  break;
210  case C_DATA:
211  result = client->msg_start (&msg, mbuf);
212  if (result == -1) {
213  logerror ("'%s': protocol error in received message\n", client_id);
214  client->state = C_PROTOCOL_ERROR;
215  goto loop;
216  } else if (result == 0) {
217  // Try again when we get more data.
218  logdebug ("'%s': need more data\n", client_id);
219  mbuf_reset_read (mbuf);
220  return;
221  } else {
222  logdebug ("'%s': received message of length %d\n", client_id, result);
223  message_length = result;
224  }
225  logdebug ("Received [strm=%d seqno=%d ts=%f %d bytes]\n",
226  msg.stream, msg.seqno, msg.timestamp, msg.length);
227 
228  mbuf_reset_read (mbuf);
229  store_received_message (client, &msg, (char*)mbuf_rdptr (mbuf), message_length);
230  mbuf_read_skip (mbuf, message_length);
231  mbuf_consume_message (mbuf);
232  break;
233  case C_PROTOCOL_ERROR:
234  logdebug ("'%s': protocol error!\n");
235  break;
236  default:
237  logerror ("'%s': unknown client state '%d'\n", client_id, client->state);
238  mbuf_clear (mbuf);
239  return;
240  }
241 }
