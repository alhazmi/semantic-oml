45 START_TEST(test_round_trip)
46 {
47  struct testcase {
48  const char *plain;
49  const char *encoded;
50  };
51 
52  static struct testcase TESTS[] = {
53  { "a", "YQ==" },
54  { "ab", "YWI=" },
55  { "abc", "YWJj" },
56  { "abcd", "YWJjZA==" },
57  { "abcde", "YWJjZGU=" },
58  { "any carnal pleasure.", "YW55IGNhcm5hbCBwbGVhc3VyZS4=" },
59  { "any carnal pleasure", "YW55IGNhcm5hbCBwbGVhc3VyZQ==" },
60  { "any carnal pleasur", "YW55IGNhcm5hbCBwbGVhc3Vy" },
61  { "any carnal pleasu", "YW55IGNhcm5hbCBwbGVhc3U=" },
62  { "any carnal pleas", "YW55IGNhcm5hbCBwbGVhcw==" },
63  };
64 
65  const int NOF_TESTS = sizeof(TESTS) / sizeof(TESTS[0]);
66 
67  const size_t MAX_PLAIN_SZ = 21;
68  char plain[MAX_PLAIN_SZ];
69  size_t plain_sz;
70  const size_t MAX_ACTUAL_SZ = MAX_PLAIN_SZ;
71  uint8_t actual[MAX_ACTUAL_SZ];
72  size_t actual_sz;
73  char *encoded = alloca(base64_size_string(MAX_PLAIN_SZ));
74  ssize_t encoded_sz;
75  size_t i;
76 
77  for(i = 0; i < NOF_TESTS; i++) {
78  /* encoding */
79  strncpy(plain, TESTS[i].plain, MAX_PLAIN_SZ);
80  plain_sz = strlen(plain);
81  encoded_sz = base64_size_string(plain_sz);
82  fail_unless(strlen(TESTS[i].encoded) + 1 == encoded_sz);
83  base64_encode_blob(plain_sz, plain, encoded);
84  fail_unless(strncmp(encoded, TESTS[i].encoded, encoded_sz) == 0, "actual=%s, expected=%s\n", encoded, TESTS[i].encoded);
85  /* decoding */
86  encoded_sz = base64_validate_string(encoded);
87  fail_if(encoded_sz < 0);
88  fail_unless('=' == encoded[encoded_sz] || '\0' == encoded[encoded_sz]);
89  actual_sz = base64_size_blob(encoded_sz);
90  fail_unless(actual_sz == strlen(TESTS[i].plain));
91  fail_if(base64_decode_string(encoded_sz, encoded, actual_sz, actual) < 0, "actual=%s, expected=%s\n", encoded, TESTS[i].plain);
92  fail_unless(memcmp(actual, TESTS[i].plain, actual_sz) == 0, "actual=%.*s, expected=%s\n", actual_sz, actual, TESTS[i].plain);
93  }
94 }
